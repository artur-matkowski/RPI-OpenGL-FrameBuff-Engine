68 inline bool RadioButtonLabeled(const char* label, bool active, bool disabled)
69 {
70  using namespace ImGui;
71 
72  ImGuiWindow* window = GetCurrentWindow();
73  if (window->SkipItems)
74  return false;
75 
76  ImGuiContext& g = *GImGui;
77  const ImGuiStyle& style = g.Style;
78  float w = CalcItemWidth();
79  if (w == window->ItemWidthDefault) w = 0.0f; // no push item width
80  const ImGuiID id = window->GetID(label);
81  const ImVec2 label_size = CalcTextSize(label, nullptr, true);
82  ImVec2 bb_size = ImVec2(style.FramePadding.x * 2 - 1, style.FramePadding.y * 2 - 1) + label_size;
83  bb_size.x = ImMax(w, bb_size.x);
84 
85  const ImRect check_bb(
86  window->DC.CursorPos,
87  window->DC.CursorPos + bb_size);
88  ItemSize(check_bb, style.FramePadding.y);
89 
90  if (!ItemAdd(check_bb, id))
91  return false;
92 
93  // check
94  bool pressed = false;
95  if (!disabled)
96  {
97  bool hovered, held;
98  pressed = ButtonBehavior(check_bb, id, &hovered, &held);
99 
100  window->DrawList->AddRectFilled(check_bb.Min, check_bb.Max, GetColorU32((held && hovered) ? ImGuiCol_FrameBgActive : hovered ? ImGuiCol_FrameBgHovered : ImGuiCol_FrameBg), style.FrameRounding);
101  if (active)
102  {
103  const ImU32 col = GetColorU32((hovered && held) ? ImGuiCol_ButtonActive : hovered ? ImGuiCol_ButtonHovered : ImGuiCol_Button);
104  window->DrawList->AddRectFilled(check_bb.Min, check_bb.Max, col, style.FrameRounding);
105  }
106  }
107 
108  // circle shadow + bg
109  if (style.FrameBorderSize > 0.0f)
110  {
111  window->DrawList->AddRect(check_bb.Min + ImVec2(1, 1), check_bb.Max, GetColorU32(ImGuiCol_BorderShadow), style.FrameRounding);
112  window->DrawList->AddRect(check_bb.Min, check_bb.Max, GetColorU32(ImGuiCol_Border), style.FrameRounding);
113  }
114 
115  if (label_size.x > 0.0f)
116  {
117  RenderText(check_bb.GetCenter() - label_size * 0.5f, label);
118  }
119 
120  return pressed;
121 }
