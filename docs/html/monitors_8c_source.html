58 static const char* format_mode(const GLFWvidmode* mode)
59 {
60  static char buffer[512];
61  const int gcd = euclid(mode->width, mode->height);
62 
63  snprintf(buffer,
64  sizeof(buffer),
65  "%i x %i x %i (%i:%i) (%i %i %i) %i Hz",
66  mode->width, mode->height,
67  mode->redBits + mode->greenBits + mode->blueBits,
68  mode->width / gcd, mode->height / gcd,
69  mode->redBits, mode->greenBits, mode->blueBits,
70  mode->refreshRate);
71 
72  buffer[sizeof(buffer) - 1] = '\0';
73  return buffer;
74 }

94 static void list_modes(GLFWmonitor* monitor)
95 {
96  int count, x, y, width_mm, height_mm, i;
97  int workarea_x, workarea_y, workarea_width, workarea_height;
98  float xscale, yscale;
99 
100  const GLFWvidmode* mode = glfwGetVideoMode(monitor);
101  const GLFWvidmode* modes = glfwGetVideoModes(monitor, &count);
102 
103  glfwGetMonitorPos(monitor, &x, &y);
104  glfwGetMonitorPhysicalSize(monitor, &width_mm, &height_mm);
105  glfwGetMonitorContentScale(monitor, &xscale, &yscale);
106  glfwGetMonitorWorkarea(monitor, &workarea_x, &workarea_y, &workarea_width, &workarea_height);
107 
108  printf("Name: %s (%s)\n",
109  glfwGetMonitorName(monitor),
110  glfwGetPrimaryMonitor() == monitor ? "primary" : "secondary");
111  printf("Current mode: %s\n", format_mode(mode));
112  printf("Virtual position: %i, %i\n", x, y);
113  printf("Content scale: %f x %f\n", xscale, yscale);
114 
115  printf("Physical size: %i x %i mm (%0.2f dpi at %i x %i)\n",
116  width_mm, height_mm, mode->width * 25.4f / width_mm, mode->width, mode->height);
117  printf("Monitor work area: %i x %i starting at %i, %i\n",
118  workarea_width, workarea_height, workarea_x, workarea_y);
119 
120  printf("Modes:\n");
121 
122  for (i = 0; i < count; i++)
123  {
124  printf("%3u: %s", (unsigned int) i, format_mode(modes + i));
125 
126  if (memcmp(mode, modes + i, sizeof(GLFWvidmode)) == 0)
127  printf(" (current mode)");
128 
129  putchar('\n');
130  }
131 }

133 static void test_modes(GLFWmonitor* monitor)
134 {
135  int i, count;
136  GLFWwindow* window;
137  const GLFWvidmode* modes = glfwGetVideoModes(monitor, &count);
138 
139  for (i = 0; i < count; i++)
140  {
141  const GLFWvidmode* mode = modes + i;
142  GLFWvidmode current;
143 
144  glfwWindowHint(GLFW_RED_BITS, mode->redBits);
145  glfwWindowHint(GLFW_GREEN_BITS, mode->greenBits);
146  glfwWindowHint(GLFW_BLUE_BITS, mode->blueBits);
147  glfwWindowHint(GLFW_REFRESH_RATE, mode->refreshRate);
148 
149  printf("Testing mode %u on monitor %s: %s\n",
150  (unsigned int) i,
151  glfwGetMonitorName(monitor),
152  format_mode(mode));
153 
154  window = glfwCreateWindow(mode->width, mode->height,
155  "Video Mode Test",
156  glfwGetPrimaryMonitor(),
157  NULL);
158  if (!window)
159  {
160  printf("Failed to enter mode %u: %s\n",
161  (unsigned int) i,
162  format_mode(mode));
163  continue;
164  }
165 
166  glfwSetFramebufferSizeCallback(window, framebuffer_size_callback);
167  glfwSetKeyCallback(window, key_callback);
168 
169  glfwMakeContextCurrent(window);
170  gladLoadGL(glfwGetProcAddress);
171  glfwSwapInterval(1);
172 
173  glfwSetTime(0.0);
174 
175  while (glfwGetTime() < 5.0)
176  {
177  glClear(GL_COLOR_BUFFER_BIT);
178  glfwSwapBuffers(window);
179  glfwPollEvents();
180 
181  if (glfwWindowShouldClose(window))
182  {
183  printf("User terminated program\n");
184 
185  glfwTerminate();
186  exit(EXIT_SUCCESS);
187  }
188  }
189 
190  glGetIntegerv(GL_RED_BITS, &current.redBits);
191  glGetIntegerv(GL_GREEN_BITS, &current.greenBits);
192  glGetIntegerv(GL_BLUE_BITS, &current.blueBits);
193 
194  glfwGetWindowSize(window, &current.width, &current.height);
195 
196  if (current.redBits != mode->redBits ||
197  current.greenBits != mode->greenBits ||
198  current.blueBits != mode->blueBits)
199  {
200  printf("*** Color bit mismatch: (%i %i %i) instead of (%i %i %i)\n",
201  current.redBits, current.greenBits, current.blueBits,
202  mode->redBits, mode->greenBits, mode->blueBits);
203  }
204 
205  if (current.width != mode->width || current.height != mode->height)
206  {
207  printf("*** Size mismatch: %ix%i instead of %ix%i\n",
208  current.width, current.height,
209  mode->width, mode->height);
210  }
211 
212  printf("Closing window\n");
213 
214  glfwDestroyWindow(window);
215  window = NULL;
216 
217  glfwPollEvents();
218  }
219 }

221 int main(int argc, char** argv)
222 {
223  int ch, i, count, mode = LIST_MODE;
224  GLFWmonitor** monitors;
225 
226  while ((ch = getopt(argc, argv, "th")) != -1)
227  {
228  switch (ch)

231  usage();
232  exit(EXIT_SUCCESS);
233  case 't':
234  mode = TEST_MODE;
235  break;
236  default:
237  usage();
238  exit(EXIT_FAILURE);
239  }
240  }
241 
242  glfwSetErrorCallback(error_callback);
243 
244  if (!glfwInit())
245  exit(EXIT_FAILURE);
246 
247  monitors = glfwGetMonitors(&count);
248 
249  for (i = 0; i < count; i++)
250  {
251  if (mode == LIST_MODE)
252  list_modes(monitors[i]);
253  else if (mode == TEST_MODE)
254  test_modes(monitors[i]);
255  }
256 
257  glfwTerminate();
258  exit(EXIT_SUCCESS);
259 }
