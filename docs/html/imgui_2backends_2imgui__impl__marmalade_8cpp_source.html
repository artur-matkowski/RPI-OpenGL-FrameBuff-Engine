42 void ImGui_Marmalade_RenderDrawData(ImDrawData* draw_data)
43 {
44  // Avoid rendering when minimized
45  if (draw_data->DisplaySize.x <= 0.0f || draw_data->DisplaySize.y <= 0.0f)
46  return;
47 
48  // Render command lists
49  for (int n = 0; n < draw_data->CmdListsCount; n++)
50  {
51  const ImDrawList* cmd_list = draw_data->CmdLists[n];
52  const ImDrawIdx* idx_buffer = cmd_list->IdxBuffer.Data;
53  const int nVert = cmd_list->VtxBuffer.Size;
54  CIwFVec2* pVertStream = IW_GX_ALLOC(CIwFVec2, nVert);
55  CIwFVec2* pUVStream = IW_GX_ALLOC(CIwFVec2, nVert);
56  CIwColour* pColStream = IW_GX_ALLOC(CIwColour, nVert);
57 
58  for (int i = 0; i < nVert; i++)
59  {
60  // FIXME-OPT: optimize multiplication on GPU using vertex shader/projection matrix.
61  pVertStream[i].x = cmd_list->VtxBuffer[i].pos.x * g_RenderScale.x;
62  pVertStream[i].y = cmd_list->VtxBuffer[i].pos.y * g_RenderScale.y;
63  pUVStream[i].x = cmd_list->VtxBuffer[i].uv.x;
64  pUVStream[i].y = cmd_list->VtxBuffer[i].uv.y;
65  pColStream[i] = cmd_list->VtxBuffer[i].col;
66  }
67 
68  IwGxSetVertStreamScreenSpace(pVertStream, nVert);
69  IwGxSetUVStream(pUVStream);
70  IwGxSetColStream(pColStream, nVert);
71  IwGxSetNormStream(0);
72 
73  for (int cmd_i = 0; cmd_i < cmd_list->CmdBuffer.Size; cmd_i++)
74  {
75  const ImDrawCmd* pcmd = &cmd_list->CmdBuffer[cmd_i];
76  if (pcmd->UserCallback)
77  {
78  pcmd->UserCallback(cmd_list, pcmd);
79  }
80  else
81  {
82  // FIXME: Not honoring ClipRect fields.
83  CIwMaterial* pCurrentMaterial = IW_GX_ALLOC_MATERIAL();
84  pCurrentMaterial->SetShadeMode(CIwMaterial::SHADE_FLAT);
85  pCurrentMaterial->SetCullMode(CIwMaterial::CULL_NONE);
86  pCurrentMaterial->SetFiltering(false);
87  pCurrentMaterial->SetAlphaMode(CIwMaterial::ALPHA_BLEND);
88  pCurrentMaterial->SetDepthWriteMode(CIwMaterial::DEPTH_WRITE_NORMAL);
89  pCurrentMaterial->SetAlphaTestMode(CIwMaterial::ALPHATEST_DISABLED);
90  pCurrentMaterial->SetTexture((CIwTexture*)pcmd->TextureId);
91  IwGxSetMaterial(pCurrentMaterial);
92  IwGxDrawPrims(IW_GX_TRI_LIST, (uint16*)idx_buffer, pcmd->ElemCount);
93  }
94  idx_buffer += pcmd->ElemCount;
95  }
96  IwGxFlush();
97  }
98 
99  // TODO: restore modified state (i.e. mvp matrix)
100 }

125 int32 ImGui_Marmalade_PointerButtonEventCallback(void* system_data, void* user_data)
126 {
127  // pEvent->m_Button is of type s3ePointerButton and indicates which mouse
128  // button was pressed. For touchscreen this should always have the value
129  // S3E_POINTER_BUTTON_SELECT
130  s3ePointerEvent* pEvent = (s3ePointerEvent*)system_data;
131 
132  if (pEvent->m_Pressed == 1)
133  {
134  if (pEvent->m_Button == S3E_POINTER_BUTTON_LEFTMOUSE)
135  g_MousePressed[0] = true;
136  if (pEvent->m_Button == S3E_POINTER_BUTTON_RIGHTMOUSE)
137  g_MousePressed[1] = true;
138  if (pEvent->m_Button == S3E_POINTER_BUTTON_MIDDLEMOUSE)
139  g_MousePressed[2] = true;
140  if (pEvent->m_Button == S3E_POINTER_BUTTON_MOUSEWHEELUP)
141  io.MouseWheel += pEvent->m_y;
142  if (pEvent->m_Button == S3E_POINTER_BUTTON_MOUSEWHEELDOWN)
143  io.MouseWheel += pEvent->m_y;
144  }
145 
146  return 0;
147 }

149 int32 ImGui_Marmalade_KeyCallback(void* system_data, void* user_data)
150 {
151  ImGuiIO& io = ImGui::GetIO();
152  s3eKeyboardEvent* e = (s3eKeyboardEvent*)system_data;
153  if (e->m_Pressed == 1)
154  io.KeysDown[e->m_Key] = true;
155  if (e->m_Pressed == 0)
156  io.KeysDown[e->m_Key] = false;
157 
158  io.KeyCtrl = s3eKeyboardGetState(s3eKeyLeftControl) == S3E_KEY_STATE_DOWN || s3eKeyboardGetState(s3eKeyRightControl) == S3E_KEY_STATE_DOWN;
159  io.KeyShift = s3eKeyboardGetState(s3eKeyLeftShift) == S3E_KEY_STATE_DOWN || s3eKeyboardGetState(s3eKeyRightShift) == S3E_KEY_STATE_DOWN;
160  io.KeyAlt = s3eKeyboardGetState(s3eKeyLeftAlt) == S3E_KEY_STATE_DOWN || s3eKeyboardGetState(s3eKeyRightAlt) == S3E_KEY_STATE_DOWN;
161  io.KeySuper = s3eKeyboardGetState(s3eKeyLeftWindows) == S3E_KEY_STATE_DOWN || s3eKeyboardGetState(s3eKeyRightWindows) == S3E_KEY_STATE_DOWN;
162

175 bool ImGui_Marmalade_CreateDeviceObjects()
176 {
177  // Build texture atlas
178  ImGuiIO& io = ImGui::GetIO();
179  unsigned char* pixels;
180  int width, height;
181  io.Fonts->GetTexDataAsRGBA32(&pixels, &width, &height);
182 
183  // Upload texture to graphics system
184  g_FontTexture = new CIwTexture();
185  g_FontTexture->SetModifiable(true);
186  CIwImage& image = g_FontTexture->GetImage();
187  image.SetFormat(CIwImage::ARGB_8888);
188  image.SetWidth(width);
189  image.SetHeight(height);
190  image.SetBuffers(); // allocates and own buffers
191  image.ReadTexels(pixels);
192  g_FontTexture->SetMipMapping(false);
193  g_FontTexture->SetFiltering(false);
194  g_FontTexture->Upload();
195 
196  // Store our identifier
197  io.Fonts->SetTexID((ImTextureID)g_FontTexture);
198 
199  return true;
200 }

264 void ImGui_Marmalade_NewFrame()
265 {
266  if (!g_FontTexture)
267  ImGui_Marmalade_CreateDeviceObjects();
268 
269  ImGuiIO& io = ImGui::GetIO();
270 
271  // Setup display size (every frame to accommodate for window resizing)
272  int w = IwGxGetScreenWidth(), h = IwGxGetScreenHeight();
273  io.DisplaySize = ImVec2((float)w, (float)h);
274  // For retina display or other situations where window coordinates are different from framebuffer coordinates. User storage only, presently not used by ImGui.
275  io.DisplayFramebufferScale = g_scale;
276 
277  // Setup time step
278  double current_time = s3eTimerGetUST() / 1000.0f;
279  io.DeltaTime = g_Time > 0.0 ? (float)(current_time - g_Time) : (float)(1.0f / 60.0f);
280  g_Time = current_time;
281 
282  double mouse_x, mouse_y;
283  mouse_x = s3ePointerGetX();
284  mouse_y = s3ePointerGetY();
285  io.MousePos = ImVec2((float)mouse_x / g_scale.x, (float)mouse_y / g_scale.y); // Mouse position (set to -FLT_MAX,-FLT_MAX if no mouse / on another screen, etc.)
286 
287  for (int i = 0; i < 3; i++)
288  {
289  io.MouseDown[i] = g_MousePressed[i] || s3ePointerGetState((s3ePointerButton)i) != S3E_POINTER_STATE_UP; // If a mouse press event came, always pass it as "mouse held this frame", so we don't miss click-release events that are shorter than 1 frame.
290  g_MousePressed[i] = false;
291  }
292 
293  // TODO: Hide OS mouse cursor if ImGui is drawing it
294  // s3ePointerSetInt(S3E_POINTER_HIDE_CURSOR,(io.MouseDrawCursor ? 0 : 1));
295 
296  // Show/hide OSD keyboard
297  if (io.WantTextInput)
298  {
299  // Some text input widget is active?
300  if (!g_osdKeyboardEnabled)
301  {
302  g_osdKeyboardEnabled = true;
303  s3eKeyboardSetInt(S3E_KEYBOARD_GET_CHAR, 1); // show OSD keyboard
304  }
305  }
306  else
307  {
308  // No text input widget is active
309  if (g_osdKeyboardEnabled)
310  {
311  g_osdKeyboardEnabled = false;
312  s3eKeyboardSetInt(S3E_KEYBOARD_GET_CHAR, 0); // hide OSD keyboard
313  }
314  }
