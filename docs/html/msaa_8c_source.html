179  {
180  float ratio;

183  const double angle = glfwGetTime() * M_PI / 180.0;
184

192

194 
195  mat4x4_translate(m, -1.f, 0.f, 0.f);
196  mat4x4_rotate_Z(m, m, (float) angle);
197  mat4x4_mul(mvp, p, m);

199  glUniformMatrix4fv(mvp_location, 1, GL_FALSE, (const GLfloat*) mvp);
200  glDisable(GL_MULTISAMPLE);

202 
203  mat4x4_translate(m, 1.f, 0.f, 0.f);
204  mat4x4_rotate_Z(m, m, (float) angle);
205  mat4x4_mul(mvp, p, m);
206

208  glEnable(GL_MULTISAMPLE);
