44 GLFWbool _glfwInitVulkan(int mode)
45 {
46  VkResult err;
47  VkExtensionProperties* ep;
48  uint32_t i, count;
49 
50  if (_glfw.vk.available)
51  return GLFW_TRUE;
52 
53 #if !defined(_GLFW_VULKAN_STATIC)
54 #if defined(_GLFW_VULKAN_LIBRARY)
55  _glfw.vk.handle = _glfw_dlopen(_GLFW_VULKAN_LIBRARY);
56 #elif defined(_GLFW_WIN32)
57  _glfw.vk.handle = _glfw_dlopen("vulkan-1.dll");
58 #elif defined(_GLFW_COCOA)
59  _glfw.vk.handle = _glfw_dlopen("libvulkan.1.dylib");
60 #else
61  _glfw.vk.handle = _glfw_dlopen("libvulkan.so.1");
62 #endif
63  if (!_glfw.vk.handle)
64  {
65  if (mode == _GLFW_REQUIRE_LOADER)
66  _glfwInputError(GLFW_API_UNAVAILABLE, "Vulkan: Loader not found");
67 
68  return GLFW_FALSE;
69  }
70 
71  _glfw.vk.GetInstanceProcAddr = (PFN_vkGetInstanceProcAddr)
72  _glfw_dlsym(_glfw.vk.handle, "vkGetInstanceProcAddr");
73  if (!_glfw.vk.GetInstanceProcAddr)
74  {
75  _glfwInputError(GLFW_API_UNAVAILABLE,
76  "Vulkan: Loader does not export vkGetInstanceProcAddr");
77 
78  _glfwTerminateVulkan();
79  return GLFW_FALSE;
80  }
81 
82  _glfw.vk.EnumerateInstanceExtensionProperties = (PFN_vkEnumerateInstanceExtensionProperties)
83  vkGetInstanceProcAddr(NULL, "vkEnumerateInstanceExtensionProperties");
84  if (!_glfw.vk.EnumerateInstanceExtensionProperties)
85  {
86  _glfwInputError(GLFW_API_UNAVAILABLE,
87  "Vulkan: Failed to retrieve vkEnumerateInstanceExtensionProperties");
88 
89  _glfwTerminateVulkan();
90  return GLFW_FALSE;
91  }
92 #endif // _GLFW_VULKAN_STATIC
93 
94  err = vkEnumerateInstanceExtensionProperties(NULL, &count, NULL);
95  if (err)
96  {
97  // NOTE: This happens on systems with a loader but without any Vulkan ICD
98  if (mode == _GLFW_REQUIRE_LOADER)
99  {
100  _glfwInputError(GLFW_API_UNAVAILABLE,
101  "Vulkan: Failed to query instance extension count: %s",
102  _glfwGetVulkanResultString(err));
103  }
104 
105  _glfwTerminateVulkan();
106  return GLFW_FALSE;
107  }
108 
109  ep = calloc(count, sizeof(VkExtensionProperties));
110 
111  err = vkEnumerateInstanceExtensionProperties(NULL, &count, ep);
112  if (err)
113  {
114  _glfwInputError(GLFW_API_UNAVAILABLE,
115  "Vulkan: Failed to query instance extensions: %s",
116  _glfwGetVulkanResultString(err));
117 
118  free(ep);
119  _glfwTerminateVulkan();
120  return GLFW_FALSE;
121  }
122 
123  for (i = 0; i < count; i++)
124  {
125  if (strcmp(ep[i].extensionName, "VK_KHR_surface") == 0)
126  _glfw.vk.KHR_surface = GLFW_TRUE;
127 #if defined(_GLFW_WIN32)
128  else if (strcmp(ep[i].extensionName, "VK_KHR_win32_surface") == 0)
129  _glfw.vk.KHR_win32_surface = GLFW_TRUE;
130 #elif defined(_GLFW_COCOA)
131  else if (strcmp(ep[i].extensionName, "VK_MVK_macos_surface") == 0)
132  _glfw.vk.MVK_macos_surface = GLFW_TRUE;
133 #elif defined(_GLFW_X11)
134  else if (strcmp(ep[i].extensionName, "VK_KHR_xlib_surface") == 0)
135  _glfw.vk.KHR_xlib_surface = GLFW_TRUE;
136  else if (strcmp(ep[i].extensionName, "VK_KHR_xcb_surface") == 0)
137  _glfw.vk.KHR_xcb_surface = GLFW_TRUE;
138 #elif defined(_GLFW_WAYLAND)
139  else if (strcmp(ep[i].extensionName, "VK_KHR_wayland_surface") == 0)
140  _glfw.vk.KHR_wayland_surface = GLFW_TRUE;
141 #endif
142  }
143 
144  free(ep);
145 
146  _glfw.vk.available = GLFW_TRUE;
147 
148  _glfwPlatformGetRequiredInstanceExtensions(_glfw.vk.extensions);
149 
150  return GLFW_TRUE;
151 }

161 const char* _glfwGetVulkanResultString(VkResult result)
162 {
163  switch (result)
164  {
165  case VK_SUCCESS:
166  return "Success";
167  case VK_NOT_READY:
168  return "A fence or query has not yet completed";
169  case VK_TIMEOUT:
170  return "A wait operation has not completed in the specified time";
171  case VK_EVENT_SET:
172  return "An event is signaled";
173  case VK_EVENT_RESET:
174  return "An event is unsignaled";
175  case VK_INCOMPLETE:
176  return "A return array was too small for the result";
177  case VK_ERROR_OUT_OF_HOST_MEMORY:
178  return "A host memory allocation has failed";
179  case VK_ERROR_OUT_OF_DEVICE_MEMORY:
180  return "A device memory allocation has failed";
181  case VK_ERROR_INITIALIZATION_FAILED:
182  return "Initialization of an object could not be completed for implementation-specific reasons";
183  case VK_ERROR_DEVICE_LOST:
184  return "The logical or physical device has been lost";
185  case VK_ERROR_MEMORY_MAP_FAILED:
186  return "Mapping of a memory object has failed";
187  case VK_ERROR_LAYER_NOT_PRESENT:
188  return "A requested layer is not present or could not be loaded";
189  case VK_ERROR_EXTENSION_NOT_PRESENT:
190  return "A requested extension is not supported";
191  case VK_ERROR_FEATURE_NOT_PRESENT:
192  return "A requested feature is not supported";
193  case VK_ERROR_INCOMPATIBLE_DRIVER:
194  return "The requested version of Vulkan is not supported by the driver or is otherwise incompatible";
195  case VK_ERROR_TOO_MANY_OBJECTS:
196  return "Too many objects of the type have already been created";
197  case VK_ERROR_FORMAT_NOT_SUPPORTED:
198  return "A requested format is not supported on this device";
199  case VK_ERROR_SURFACE_LOST_KHR:
200  return "A surface is no longer available";
201  case VK_SUBOPTIMAL_KHR:
202  return "A swapchain no longer matches the surface properties exactly, but can still be used";
203  case VK_ERROR_OUT_OF_DATE_KHR:
204  return "A surface has changed in such a way that it is no longer compatible with the swapchain";
205  case VK_ERROR_INCOMPATIBLE_DISPLAY_KHR:
206  return "The display used by a swapchain does not use the same presentable image layout";
207  case VK_ERROR_NATIVE_WINDOW_IN_USE_KHR:
208  return "The requested window is already connected to a VkSurfaceKHR, or to some other non-Vulkan API";
209  case VK_ERROR_VALIDATION_FAILED_EXT:
210  return "A validation layer found an error";
211  default:
212  return "ERROR: UNKNOWN VULKAN ERROR";
213  }
