256 static size_t encode_utf8(char* s, unsigned int ch)
257 {
258  size_t count = 0;
259 
260  if (ch < 0x80)
261  s[count++] = (char) ch;
262  else if (ch < 0x800)
263  {
264  s[count++] = (ch >> 6) | 0xc0;
265  s[count++] = (ch & 0x3f) | 0x80;
266  }
267  else if (ch < 0x10000)
268  {
269  s[count++] = (ch >> 12) | 0xe0;
270  s[count++] = ((ch >> 6) & 0x3f) | 0x80;
271  s[count++] = (ch & 0x3f) | 0x80;
272  }
273  else if (ch < 0x110000)
274  {
275  s[count++] = (ch >> 18) | 0xf0;
276  s[count++] = ((ch >> 12) & 0x3f) | 0x80;
277  s[count++] = ((ch >> 6) & 0x3f) | 0x80;
278  s[count++] = (ch & 0x3f) | 0x80;
279  }
280 
281  return count;
282 }

399 static void key_callback(GLFWwindow* window, int key, int scancode, int action, int mods)
400 {
401  Slot* slot = glfwGetWindowUserPointer(window);
402  const char* name = glfwGetKeyName(key, scancode);
403 
404  if (name)
405  {
406  printf("%08x to %i at %0.3f: Key 0x%04x Scancode 0x%04x (%s) (%s) (with%s) was %s\n",
407  counter++, slot->number, glfwGetTime(), key, scancode,
408  get_key_name(key),
409  name,
410  get_mods_name(mods),
411  get_action_name(action));
412  }
413  else
414  {
415  printf("%08x to %i at %0.3f: Key 0x%04x Scancode 0x%04x (%s) (with%s) was %s\n",
416  counter++, slot->number, glfwGetTime(), key, scancode,
417  get_key_name(key),
418  get_mods_name(mods),
419  get_action_name(action));
420  }
421 
422  if (action != GLFW_PRESS)
423  return;
424 
425  switch (key)
426  {
427  case GLFW_KEY_C:
428  {
429  slot->closeable = !slot->closeable;
430 
431  printf("(( closing %s ))\n", slot->closeable ? "enabled" : "disabled");
432  break;
433  }
434 
435  case GLFW_KEY_L:
436  {
437  const int state = glfwGetInputMode(window, GLFW_LOCK_KEY_MODS);
438  glfwSetInputMode(window, GLFW_LOCK_KEY_MODS, !state);
439 
440  printf("(( lock key mods %s ))\n", !state ? "enabled" : "disabled");
441  break;
442  }
443  }
444 }

468 static void monitor_callback(GLFWmonitor* monitor, int event)
469 {
470  if (event == GLFW_CONNECTED)
471  {
472  int x, y, widthMM, heightMM;
473  const GLFWvidmode* mode = glfwGetVideoMode(monitor);
474 
475  glfwGetMonitorPos(monitor, &x, &y);
476  glfwGetMonitorPhysicalSize(monitor, &widthMM, &heightMM);
477 
478  printf("%08x at %0.3f: Monitor %s (%ix%i at %ix%i, %ix%i mm) was connected\n",
479  counter++,
480  glfwGetTime(),
481  glfwGetMonitorName(monitor),
482  mode->width, mode->height,
483  x, y,
484  widthMM, heightMM);
485  }
486  else if (event == GLFW_DISCONNECTED)
487  {
488  printf("%08x at %0.3f: Monitor %s was disconnected\n",
489  counter++,
490  glfwGetTime(),
491  glfwGetMonitorName(monitor));
492  }
493 }

495 static void joystick_callback(int jid, int event)
496 {
497  if (event == GLFW_CONNECTED)
498  {
499  int axisCount, buttonCount, hatCount;
500 
501  glfwGetJoystickAxes(jid, &axisCount);
502  glfwGetJoystickButtons(jid, &buttonCount);
503  glfwGetJoystickHats(jid, &hatCount);
504 
505  printf("%08x at %0.3f: Joystick %i (%s) was connected with %i axes, %i buttons, and %i hats\n",
506  counter++, glfwGetTime(),
507  jid,
508  glfwGetJoystickName(jid),
509  axisCount,
510  buttonCount,
511  hatCount);
512 
513  if (glfwJoystickIsGamepad(jid))
514  {
515  printf(" Joystick %i (%s) has a gamepad mapping (%s)\n",
516  jid,
517  glfwGetJoystickGUID(jid),
518  glfwGetGamepadName(jid));
519  }
520  else
521  {
522  printf(" Joystick %i (%s) has no gamepad mapping\n",
523  jid,
524  glfwGetJoystickGUID(jid));
525  }
526  }
527  else
528  {
529  printf("%08x at %0.3f: Joystick %i was disconnected\n",
530  counter++, glfwGetTime(), jid);
531  }
532 }

534 int main(int argc, char** argv)
535 {
536  Slot* slots;
537  GLFWmonitor* monitor = NULL;
538  int ch, i, width, height, count = 1;
539 
540  glfwSetErrorCallback(error_callback);
541 
542  if (!glfwInit())
543  exit(EXIT_FAILURE);
544 
545  printf("Library initialized\n");
546 
547  glfwSetMonitorCallback(monitor_callback);
548  glfwSetJoystickCallback(joystick_callback);
549 
550  while ((ch = getopt(argc, argv, "hfn:")) != -1)
551  {
552  switch (ch)
553  {
554  case 'h':
555  usage();
556  exit(EXIT_SUCCESS);
557 
558  case 'f':
559  monitor = glfwGetPrimaryMonitor();
560  break;
561 
562  case 'n':
563  count = (int) strtoul(optarg, NULL, 10);
564  break;
565 
566  default:
567  usage();
568  exit(EXIT_FAILURE);
569  }
570  }
571 
572  if (monitor)
573  {
574  const GLFWvidmode* mode = glfwGetVideoMode(monitor);
575 
576  glfwWindowHint(GLFW_REFRESH_RATE, mode->refreshRate);
577  glfwWindowHint(GLFW_RED_BITS, mode->redBits);
578  glfwWindowHint(GLFW_GREEN_BITS, mode->greenBits);
579  glfwWindowHint(GLFW_BLUE_BITS, mode->blueBits);
580 
581  width = mode->width;
582  height = mode->height;
583  }
584  else
585  {
586  width = 640;
587  height = 480;
588  }
589 
590  slots = calloc(count, sizeof(Slot));
591 
592  for (i = 0; i < count; i++)
593  {
594  char title[128];
595 
596  slots[i].closeable = GLFW_TRUE;
597  slots[i].number = i + 1;
598 
599  snprintf(title, sizeof(title), "Event Linter (Window %i)", slots[i].number);
600 
601  if (monitor)
602  {
603  printf("Creating full screen window %i (%ix%i on %s)\n",
604  slots[i].number,
605  width, height,
606  glfwGetMonitorName(monitor));
607  }
608  else
609  {
610  printf("Creating windowed mode window %i (%ix%i)\n",
611  slots[i].number,
612  width, height);
613  }
614 
615  slots[i].window = glfwCreateWindow(width, height, title, monitor, NULL);
616  if (!slots[i].window)
617  {
618  free(slots);
619  glfwTerminate();
620  exit(EXIT_FAILURE);
621  }
622 
623  glfwSetWindowUserPointer(slots[i].window, slots + i);
624 
625  glfwSetWindowPosCallback(slots[i].window, window_pos_callback);
626  glfwSetWindowSizeCallback(slots[i].window, window_size_callback);
627  glfwSetFramebufferSizeCallback(slots[i].window, framebuffer_size_callback);
628  glfwSetWindowContentScaleCallback(slots[i].window, window_content_scale_callback);
629  glfwSetWindowCloseCallback(slots[i].window, window_close_callback);
630  glfwSetWindowRefreshCallback(slots[i].window, window_refresh_callback);
631  glfwSetWindowFocusCallback(slots[i].window, window_focus_callback);
632  glfwSetWindowIconifyCallback(slots[i].window, window_iconify_callback);
