178 static void list_context_extensions(int client, int major, int minor)
179 {
180  printf("%s context extensions:\n", get_api_name(client));
181 
182  if (client == GLFW_OPENGL_API && major > 2)
183  {
184  GLint count;
185  glGetIntegerv(GL_NUM_EXTENSIONS, &count);
186 
187  for (int i = 0; i < count; i++)
188  printf(" %s\n", (const char*) glGetStringi(GL_EXTENSIONS, i));
189  }
190  else
191  {
192  const GLubyte* extensions = glGetString(GL_EXTENSIONS);
193  while (*extensions != '\0')
194  {
195  putchar(' ');
196 
197  while (*extensions != '\0' && *extensions != ' ')
198  {
199  putchar(*extensions);
200  extensions++;
201  }
202 
203  while (*extensions == ' ')
204  extensions++;
205 
206  putchar('\n');
207  }
208  }
209 }

281 static int valid_version(void)
282 {
283  int major, minor, revision;
284  glfwGetVersion(&major, &minor, &revision);
285 
286  if (major != GLFW_VERSION_MAJOR)
287  {
288  printf("*** ERROR: GLFW major version mismatch! ***\n");
289  return GLFW_FALSE;
290  }
291 
292  if (minor != GLFW_VERSION_MINOR || revision != GLFW_VERSION_REVISION)
293  printf("*** WARNING: GLFW version mismatch! ***\n");
294 
295  return GLFW_TRUE;
296 }
