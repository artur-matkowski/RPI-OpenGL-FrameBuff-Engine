79 // Returns the image whose area most closely matches the desired one
80 //
81 static const GLFWimage* chooseImage(int count, const GLFWimage* images,
82  int width, int height)
83 {
84  int i, leastDiff = INT_MAX;
85  const GLFWimage* closest = NULL;
86 
87  for (i = 0; i < count; i++)
88  {
89  const int currDiff = abs(images[i].width * images[i].height -
90  width * height);
91  if (currDiff < leastDiff)
92  {
93  closest = images + i;
94  leastDiff = currDiff;
95  }
96  }
97 
98  return closest;

101 // Creates an RGBA icon or cursor
102 //
103 static HICON createIcon(const GLFWimage* image,
104  int xhot, int yhot, GLFWbool icon)
105 {
106  int i;
107  HDC dc;
108  HICON handle;
109  HBITMAP color, mask;
110  BITMAPV5HEADER bi;
111  ICONINFO ii;
112  unsigned char* target = NULL;
113  unsigned char* source = image->pixels;
114 
115  ZeroMemory(&bi, sizeof(bi));
116  bi.bV5Size = sizeof(bi);
117  bi.bV5Width = image->width;
118  bi.bV5Height = -image->height;
119  bi.bV5Planes = 1;
120  bi.bV5BitCount = 32;
121  bi.bV5Compression = BI_BITFIELDS;
122  bi.bV5RedMask = 0x00ff0000;
123  bi.bV5GreenMask = 0x0000ff00;
124  bi.bV5BlueMask = 0x000000ff;
125  bi.bV5AlphaMask = 0xff000000;
126 
127  dc = GetDC(NULL);
128  color = CreateDIBSection(dc,
129  (BITMAPINFO*) &bi,
130  DIB_RGB_COLORS,
131  (void**) &target,
132  NULL,
133  (DWORD) 0);
134  ReleaseDC(NULL, dc);
135 
136  if (!color)
137  {
138  _glfwInputErrorWin32(GLFW_PLATFORM_ERROR,
139  "Win32: Failed to create RGBA bitmap");
140  return NULL;
141  }
142 
143  mask = CreateBitmap(image->width, image->height, 1, 1, NULL);
144  if (!mask)
145  {
146  _glfwInputErrorWin32(GLFW_PLATFORM_ERROR,
147  "Win32: Failed to create mask bitmap");
148  DeleteObject(color);
149  return NULL;
150  }
151 
152  for (i = 0; i < image->width * image->height; i++)
153  {
154  target[0] = source[2];
155  target[1] = source[1];
156  target[2] = source[0];
157  target[3] = source[3];
158  target += 4;
159  source += 4;
160  }
161 
162  ZeroMemory(&ii, sizeof(ii));
163  ii.fIcon = icon;
164  ii.xHotspot = xhot;
165  ii.yHotspot = yhot;
166  ii.hbmMask = mask;
167  ii.hbmColor = color;
168 
169  handle = CreateIconIndirect(&ii);
170 
171  DeleteObject(color);
172  DeleteObject(mask);
173 
174  if (!handle)
175  {
176  if (icon)
177  {
178  _glfwInputErrorWin32(GLFW_PLATFORM_ERROR,
179  "Win32: Failed to create icon");
180  }
181  else
182  {
183  _glfwInputErrorWin32(GLFW_PLATFORM_ERROR,
184  "Win32: Failed to create cursor");
185  }
186  }
187 
188  return handle;
189 }

209 // Enforce the content area aspect ratio based on which edge is being dragged
210 //
211 static void applyAspectRatio(_GLFWwindow* window, int edge, RECT* area)
212 {
213  int xoff, yoff;
214  UINT dpi = USER_DEFAULT_SCREEN_DPI;
215  const float ratio = (float) window->numer / (float) window->denom;
216 
217  if (_glfwIsWindows10AnniversaryUpdateOrGreaterWin32())
218  dpi = GetDpiForWindow(window->win32.handle);
219 
220  getFullWindowSize(getWindowStyle(window), getWindowExStyle(window),
221  0, 0, &xoff, &yoff, dpi);
222 
223  if (edge == WMSZ_LEFT || edge == WMSZ_BOTTOMLEFT ||
224  edge == WMSZ_RIGHT || edge == WMSZ_BOTTOMRIGHT)
225  {
226  area->bottom = area->top + yoff +
227  (int) ((area->right - area->left - xoff) / ratio);
228  }
229  else if (edge == WMSZ_TOPLEFT || edge == WMSZ_TOPRIGHT)
230  {
231  area->top = area->bottom - yoff -
232  (int) ((area->right - area->left - xoff) / ratio);
233  }
234  else if (edge == WMSZ_TOP || edge == WMSZ_BOTTOM)
235  {
236  area->right = area->left + xoff +
237  (int) ((area->bottom - area->top - yoff) * ratio);
238  }
239 }

378 // Update window framebuffer transparency
379 //
380 static void updateFramebufferTransparency(const _GLFWwindow* window)
381 {
382  BOOL enabled;
383 
384  if (!IsWindowsVistaOrGreater())
385  return;
386 
387  if (SUCCEEDED(DwmIsCompositionEnabled(&enabled)) && enabled)
388  {
389  HRGN region = CreateRectRgn(0, 0, -1, -1);
390  DWM_BLURBEHIND bb = {0};
391  bb.dwFlags = DWM_BB_ENABLE | DWM_BB_BLURREGION;
392  bb.hRgnBlur = region;
393  bb.fEnable = TRUE;
394 
395  if (SUCCEEDED(DwmEnableBlurBehindWindow(window->win32.handle, &bb)))
396  {
397  // Decorated windows don't repaint the transparent background
398  // leaving a trail behind animations
399  // HACK: Making the window layered with a transparency color key
400  // seems to fix this. Normally, when specifying
401  // a transparency color key to be used when composing the
402  // layered window, all pixels painted by the window in this
403  // color will be transparent. That doesn't seem to be the
404  // case anymore, at least when used with blur behind window
405  // plus negative region.
406  LONG exStyle = GetWindowLongW(window->win32.handle, GWL_EXSTYLE);
407  exStyle |= WS_EX_LAYERED;
408  SetWindowLongW(window->win32.handle, GWL_EXSTYLE, exStyle);
409 
410  // Using a color key not equal to black to fix the trailing
411  // issue. When set to black, something is making the hit test
412  // not resize with the window frame.
413  SetLayeredWindowAttributes(window->win32.handle,
414  RGB(255, 0, 255), 255, LWA_COLORKEY);
415  }
416 
417  DeleteObject(region);
418  }
419  else
420  {
421  LONG exStyle = GetWindowLongW(window->win32.handle, GWL_EXSTYLE);
422  exStyle &= ~WS_EX_LAYERED;
423  SetWindowLongW(window->win32.handle, GWL_EXSTYLE, exStyle);
424  RedrawWindow(window->win32.handle, NULL, NULL,
425  RDW_ERASE | RDW_INVALIDATE | RDW_FRAME);
426  }
427 }

473 // Translates a Windows key to the corresponding GLFW key
474 //
475 static int translateKey(WPARAM wParam, LPARAM lParam)
476 {
477  // The Ctrl keys require special handling
478  if (wParam == VK_CONTROL)
479  {
480  MSG next;
481  DWORD time;
482 
483  // Right side keys have the extended key bit set
484  if (HIWORD(lParam) & KF_EXTENDED)
485  return GLFW_KEY_RIGHT_CONTROL;
486 
487  // HACK: Alt Gr sends Left Ctrl and then Right Alt in close sequence
488  // We only want the Right Alt message, so if the next message is
489  // Right Alt we ignore this (synthetic) Left Ctrl message
490  time = GetMessageTime();
491 
492  if (PeekMessageW(&next, NULL, 0, 0, PM_NOREMOVE))
493  {
494  if (next.message == WM_KEYDOWN ||
495  next.message == WM_SYSKEYDOWN ||
496  next.message == WM_KEYUP ||
497  next.message == WM_SYSKEYUP)
498  {
499  if (next.wParam == VK_MENU &&
500  (HIWORD(next.lParam) & KF_EXTENDED) &&
501  next.time == time)
502  {
503  // Next message is Right Alt down so discard this
504  return _GLFW_KEY_INVALID;
505  }
506  }
507  }
508 
509  return GLFW_KEY_LEFT_CONTROL;
510  }
511 
512  if (wParam == VK_PROCESSKEY)
513  {
514  // IME notifies that keys have been filtered by setting the virtual
515  // key-code to VK_PROCESSKEY
516  return _GLFW_KEY_INVALID;
517  }
518 
519  return _glfw.win32.keycodes[HIWORD(lParam) & 0x1FF];
520 }

579 // Window callback function (handles window messages)
580 //
581 static LRESULT CALLBACK windowProc(HWND hWnd, UINT uMsg,
582  WPARAM wParam, LPARAM lParam)
583 {
584  _GLFWwindow* window = GetPropW(hWnd, L"GLFW");
585  if (!window)
586  {
587  // This is the message handling for the hidden helper window
588  // and for a regular window during its initial creation
589 
590  switch (uMsg)
591  {
592  case WM_NCCREATE:
593  {
594  if (_glfwIsWindows10AnniversaryUpdateOrGreaterWin32())
595  EnableNonClientDpiScaling(hWnd);
596 
597  break;
598  }
599 
600  case WM_DISPLAYCHANGE:
601  _glfwPollMonitorsWin32();
602  break;
603 
604  case WM_DEVICECHANGE:
605  {
606  if (wParam == DBT_DEVICEARRIVAL)
607  {
608  DEV_BROADCAST_HDR* dbh = (DEV_BROADCAST_HDR*) lParam;
609  if (dbh && dbh->dbch_devicetype == DBT_DEVTYP_DEVICEINTERFACE)
610  _glfwDetectJoystickConnectionWin32();
611  }
612  else if (wParam == DBT_DEVICEREMOVECOMPLETE)
613  {
614  DEV_BROADCAST_HDR* dbh = (DEV_BROADCAST_HDR*) lParam;
615  if (dbh && dbh->dbch_devicetype == DBT_DEVTYP_DEVICEINTERFACE)
616  _glfwDetectJoystickDisconnectionWin32();
617  }
618 
619  break;
620  }
621  }
622 
623  return DefWindowProcW(hWnd, uMsg, wParam, lParam);
624  }
625 
626  switch (uMsg)
627  {
628  case WM_MOUSEACTIVATE:
629  {
630  // HACK: Postpone cursor disabling when the window was activated by
631  // clicking a caption button
632  if (HIWORD(lParam) == WM_LBUTTONDOWN)
633  {
634  if (LOWORD(lParam) != HTCLIENT)
635  window->win32.frameAction = GLFW_TRUE;
636  }
637 
638  break;
639  }
640 
641  case WM_CAPTURECHANGED:
642  {
643  // HACK: Disable the cursor once the caption button action has been
644  // completed or cancelled
645  if (lParam == 0 && window->win32.frameAction)
646  {
647  if (window->cursorMode == GLFW_CURSOR_DISABLED)
648  disableCursor(window);
649 
650  window->win32.frameAction = GLFW_FALSE;
651  }
652 
653  break;
654  }
655 
656  case WM_SETFOCUS:
657  {
658  _glfwInputWindowFocus(window, GLFW_TRUE);
659 
660  // HACK: Do not disable cursor while the user is interacting with
661  // a caption button
662  if (window->win32.frameAction)
663  break;
664 
665  if (window->cursorMode == GLFW_CURSOR_DISABLED)
666  disableCursor(window);
667 
668  return 0;
669  }
670 
671  case WM_KILLFOCUS:
672  {
673  if (window->cursorMode == GLFW_CURSOR_DISABLED)
674  enableCursor(window);
675 
676  if (window->monitor && window->autoIconify)
677  _glfwPlatformIconifyWindow(window);
678 
679  _glfwInputWindowFocus(window, GLFW_FALSE);
680  return 0;
681  }
682 
683  case WM_SYSCOMMAND:
684  {
685  switch (wParam & 0xfff0)
686  {
687  case SC_SCREENSAVE:
688  case SC_MONITORPOWER:
689  {
690  if (window->monitor)
691  {
692  // We are running in full screen mode, so disallow
693  // screen saver and screen blanking
694  return 0;
695  }
696  else
697  break;
698  }
699 
700  // User trying to access application menu using ALT?
701  case SC_KEYMENU:
702  {
703  if (!window->win32.keymenu)
704  return 0;
705 
706  break;
707  }
708  }
709  break;
710  }
711 
712  case WM_CLOSE:
713  {
714  _glfwInputWindowCloseRequest(window);
715  return 0;
716  }
717 
718  case WM_INPUTLANGCHANGE:
719  {
720  _glfwUpdateKeyNamesWin32();
721  break;
722  }
723 
724  case WM_CHAR:
725  case WM_SYSCHAR:
726  case WM_UNICHAR:
727  {
728  const GLFWbool plain = (uMsg != WM_SYSCHAR);
729 
730  if (uMsg == WM_UNICHAR && wParam == UNICODE_NOCHAR)
731  {
732  // WM_UNICHAR is not sent by Windows, but is sent by some
733  // third-party input method engine

1909 void _glfwPlatformPollEvents(void)
1910 {
1911  MSG msg;
1912  HWND handle;
1913  _GLFWwindow* window;
1914 
1915  while (PeekMessageW(&msg, NULL, 0, 0, PM_REMOVE))
1916  {
1917  if (msg.message == WM_QUIT)
1918  {
1919  // NOTE: While GLFW does not itself post WM_QUIT, other processes
1920  // may post it to this one, for example Task Manager
1921  // HACK: Treat WM_QUIT as a close on all windows
1922 
1923  window = _glfw.windowListHead;
1924  while (window)
1925  {
1926  _glfwInputWindowCloseRequest(window);
1927  window = window->next;
1928  }
1929  }
1930  else
1931  {
1932  TranslateMessage(&msg);
1933  DispatchMessageW(&msg);
1934  }
1935  }
1936 
1937  handle = GetActiveWindow();
1938  if (handle)
1939  {
1940  // NOTE: Shift keys on Windows tend to "stick" when both are pressed as
1941  // no key up message is generated by the first key release
1942  // The other half of this is in the handling of WM_KEYUP
1943  // HACK: Query actual key state and synthesize release events as needed
1944  window = GetPropW(handle, L"GLFW");
1945  if (window)
1946  {
1947  const GLFWbool lshift = (GetAsyncKeyState(VK_LSHIFT) & 0x8000) != 0;
1948  const GLFWbool rshift = (GetAsyncKeyState(VK_RSHIFT) & 0x8000) != 0;
1949 
1950  if (!lshift && window->keys[GLFW_KEY_LEFT_SHIFT] == GLFW_PRESS)
1951  {
1952  const int mods = getAsyncKeyMods();
1953  const int scancode = _glfw.win32.scancodes[GLFW_KEY_LEFT_SHIFT];
1954  _glfwInputKey(window, GLFW_KEY_LEFT_SHIFT, scancode, GLFW_RELEASE, mods);
1955  }
1956  else if (!rshift && window->keys[GLFW_KEY_RIGHT_SHIFT] == GLFW_PRESS)
1957  {
1958  const int mods = getAsyncKeyMods();
1959  const int scancode = _glfw.win32.scancodes[GLFW_KEY_RIGHT_SHIFT];
1960  _glfwInputKey(window, GLFW_KEY_RIGHT_SHIFT, scancode, GLFW_RELEASE, mods);
1961  }
1962  }
1963  }
1964 
1965  window = _glfw.win32.disabledCursorWindow;
1966  if (window)
1967  {
1968  int width, height;
1969  _glfwPlatformGetWindowSize(window, &width, &height);
1970 
1971  // NOTE: Re-center the cursor only if it has moved since the last call,
1972  // to avoid breaking glfwWaitEvents with WM_MOUSEMOVE
1973  if (window->win32.lastCursorPosX != width / 2 ||
1974  window->win32.lastCursorPosY != height / 2)
1975  {
1976  _glfwPlatformSetCursorPos(window, width / 2, height / 2);
1977  }
1978  }
1979 }
