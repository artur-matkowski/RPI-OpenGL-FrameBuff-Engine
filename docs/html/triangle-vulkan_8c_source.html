585 static void demo_draw(struct demo *demo) {
586  VkResult U_ASSERT_ONLY err;
587  VkSemaphore imageAcquiredSemaphore, drawCompleteSemaphore;
588  VkSemaphoreCreateInfo semaphoreCreateInfo = {
589  .sType = VK_STRUCTURE_TYPE_SEMAPHORE_CREATE_INFO,
590  .pNext = NULL,
591  .flags = 0,
592  };
593 
594  err = vkCreateSemaphore(demo->device, &semaphoreCreateInfo,
595  NULL, &imageAcquiredSemaphore);
596  assert(!err);
597 
598  err = vkCreateSemaphore(demo->device, &semaphoreCreateInfo,
599  NULL, &drawCompleteSemaphore);
600  assert(!err);
601 
602  // Get the index of the next available swapchain image:
603  err = vkAcquireNextImageKHR(demo->device, demo->swapchain, UINT64_MAX,
604  imageAcquiredSemaphore,
605  (VkFence)0, // TODO: Show use of fence
606  &demo->current_buffer);
607  if (err == VK_ERROR_OUT_OF_DATE_KHR) {
608  // demo->swapchain is out of date (e.g. the window was resized) and
609  // must be recreated:
610  demo_resize(demo);
611  demo_draw(demo);
612  vkDestroySemaphore(demo->device, imageAcquiredSemaphore, NULL);
613  vkDestroySemaphore(demo->device, drawCompleteSemaphore, NULL);
614  return;
615  } else if (err == VK_SUBOPTIMAL_KHR) {
616  // demo->swapchain is not as optimal as it could be, but the platform's
617  // presentation engine will still present the image correctly.
618  } else {
619  assert(!err);
620  }
621 
622  demo_flush_init_cmd(demo);
623 
624  // Wait for the present complete semaphore to be signaled to ensure
625  // that the image won't be rendered to until the presentation
626  // engine has fully released ownership to the application, and it is
627  // okay to render to the image.
628 
629  demo_draw_build_cmd(demo);
630  VkFence nullFence = VK_NULL_HANDLE;
631  VkPipelineStageFlags pipe_stage_flags =
632  VK_PIPELINE_STAGE_BOTTOM_OF_PIPE_BIT;
633  VkSubmitInfo submit_info = {.sType = VK_STRUCTURE_TYPE_SUBMIT_INFO,
634  .pNext = NULL,
635  .waitSemaphoreCount = 1,
636  .pWaitSemaphores = &imageAcquiredSemaphore,
637  .pWaitDstStageMask = &pipe_stage_flags,
638  .commandBufferCount = 1,
639  .pCommandBuffers = &demo->draw_cmd,
640  .signalSemaphoreCount = 1,
641  .pSignalSemaphores = &drawCompleteSemaphore};
642 
643  err = vkQueueSubmit(demo->queue, 1, &submit_info, nullFence);
644  assert(!err);
645 
646  VkPresentInfoKHR present = {
647  .sType = VK_STRUCTURE_TYPE_PRESENT_INFO_KHR,
648  .pNext = NULL,
649  .waitSemaphoreCount = 1,
650  .pWaitSemaphores = &drawCompleteSemaphore,
651  .swapchainCount = 1,
652  .pSwapchains = &demo->swapchain,
653  .pImageIndices = &demo->current_buffer,
654  };
655 
656  err = vkQueuePresentKHR(demo->queue, &present);
657  if (err == VK_ERROR_OUT_OF_DATE_KHR) {
658  // demo->swapchain is out of date (e.g. the window was resized) and
659  // must be recreated:
660  demo_resize(demo);
661  } else if (err == VK_SUBOPTIMAL_KHR) {
662  // demo->swapchain is not as optimal as it could be, but the platform's
663  // presentation engine will still present the image correctly.
664  } else {
665  assert(!err);
666  }
667 
668  err = vkQueueWaitIdle(demo->queue);
669  assert(err == VK_SUCCESS);
670 
671  vkDestroySemaphore(demo->device, imageAcquiredSemaphore, NULL);
672  vkDestroySemaphore(demo->device, drawCompleteSemaphore, NULL);
673 }

675 static void demo_prepare_buffers(struct demo *demo) {
676  VkResult U_ASSERT_ONLY err;
677  VkSwapchainKHR oldSwapchain = demo->swapchain;
678 
679  // Check the surface capabilities and formats
680  VkSurfaceCapabilitiesKHR surfCapabilities;
681  err = vkGetPhysicalDeviceSurfaceCapabilitiesKHR(
682  demo->gpu, demo->surface, &surfCapabilities);
683  assert(!err);
684 
685  uint32_t presentModeCount;
686  err = vkGetPhysicalDeviceSurfacePresentModesKHR(
687  demo->gpu, demo->surface, &presentModeCount, NULL);
688  assert(!err);
689  VkPresentModeKHR *presentModes =
690  (VkPresentModeKHR *)malloc(presentModeCount * sizeof(VkPresentModeKHR));
691  assert(presentModes);
692  err = vkGetPhysicalDeviceSurfacePresentModesKHR(
693  demo->gpu, demo->surface, &presentModeCount, presentModes);
694  assert(!err);
695 
696  VkExtent2D swapchainExtent;
697  // width and height are either both 0xFFFFFFFF, or both not 0xFFFFFFFF.
698  if (surfCapabilities.currentExtent.width == 0xFFFFFFFF) {
699  // If the surface size is undefined, the size is set to the size
700  // of the images requested, which must fit within the minimum and
701  // maximum values.
702  swapchainExtent.width = demo->width;
703  swapchainExtent.height = demo->height;
704 
705  if (swapchainExtent.width < surfCapabilities.minImageExtent.width) {
706  swapchainExtent.width = surfCapabilities.minImageExtent.width;
707  } else if (swapchainExtent.width > surfCapabilities.maxImageExtent.width) {
708  swapchainExtent.width = surfCapabilities.maxImageExtent.width;
709  }
710 
711  if (swapchainExtent.height < surfCapabilities.minImageExtent.height) {
712  swapchainExtent.height = surfCapabilities.minImageExtent.height;
713  } else if (swapchainExtent.height > surfCapabilities.maxImageExtent.height) {
714  swapchainExtent.height = surfCapabilities.maxImageExtent.height;
715  }
716  } else {
717  // If the surface size is defined, the swap chain size must match
718  swapchainExtent = surfCapabilities.currentExtent;
719  demo->width = surfCapabilities.currentExtent.width;
720  demo->height = surfCapabilities.currentExtent.height;
721  }
722 
723  VkPresentModeKHR swapchainPresentMode = VK_PRESENT_MODE_FIFO_KHR;
724 
725  // Determine the number of VkImage's to use in the swap chain.
726  // Application desires to only acquire 1 image at a time (which is
727  // "surfCapabilities.minImageCount").
728  uint32_t desiredNumOfSwapchainImages = surfCapabilities.minImageCount;
729  // If maxImageCount is 0, we can ask for as many images as we want;
730  // otherwise we're limited to maxImageCount
731  if ((surfCapabilities.maxImageCount > 0) &&
732  (desiredNumOfSwapchainImages > surfCapabilities.maxImageCount)) {
733  // Application must settle for fewer images than desired:
734  desiredNumOfSwapchainImages = surfCapabilities.maxImageCount;
735  }
736 
737  VkSurfaceTransformFlagsKHR preTransform;
738  if (surfCapabilities.supportedTransforms &
739  VK_SURFACE_TRANSFORM_IDENTITY_BIT_KHR) {
740  preTransform = VK_SURFACE_TRANSFORM_IDENTITY_BIT_KHR;
741  } else {
742  preTransform = surfCapabilities.currentTransform;
743  }
744 
745  const VkSwapchainCreateInfoKHR swapchain = {
746  .sType = VK_STRUCTURE_TYPE_SWAPCHAIN_CREATE_INFO_KHR,
747  .pNext = NULL,
748  .surface = demo->surface,
749  .minImageCount = desiredNumOfSwapchainImages,
750  .imageFormat = demo->format,
751  .imageColorSpace = demo->color_space,
752  .imageExtent =
753  {
754  .width = swapchainExtent.width, .height = swapchainExtent.height,
755  },
756  .imageUsage = VK_IMAGE_USAGE_COLOR_ATTACHMENT_BIT,
757  .preTransform = preTransform,
758  .compositeAlpha = VK_COMPOSITE_ALPHA_OPAQUE_BIT_KHR,
759  .imageArrayLayers = 1,
760  .imageSharingMode = VK_SHARING_MODE_EXCLUSIVE,
761  .queueFamilyIndexCount = 0,
762  .pQueueFamilyIndices = NULL,
763  .presentMode = swapchainPresentMode,
764  .oldSwapchain = oldSwapchain,
765  .clipped = true,
766  };
767  uint32_t i;
768 
769  err = vkCreateSwapchainKHR(demo->device, &swapchain, NULL, &demo->swapchain);
770  assert(!err);
771 
772  // If we just re-created an existing swapchain, we should destroy the old
773  // swapchain at this point.
774  // Note: destroying the swapchain also cleans up all its associated
775  // presentable images once the platform is done with them.
776  if (oldSwapchain != VK_NULL_HANDLE) {
777  vkDestroySwapchainKHR(demo->device, oldSwapchain, NULL);
778  }
779 
780  err = vkGetSwapchainImagesKHR(demo->device, demo->swapchain,
781  &demo->swapchainImageCount, NULL);
782  assert(!err);
783 
784  VkImage *swapchainImages =
785  (VkImage *)malloc(demo->swapchainImageCount * sizeof(VkImage));
786  assert(swapchainImages);
787  err = vkGetSwapchainImagesKHR(demo->device, demo->swapchain,
788  &demo->swapchainImageCount,
789  swapchainImages);
790  assert(!err);
791 
792  demo->buffers = (SwapchainBuffers *)malloc(sizeof(SwapchainBuffers) *
793  demo->swapchainImageCount);
794  assert(demo->buffers);
