172 static bool ImGui_ImplWin32_UpdateMouseCursor()
173 {
174  ImGuiIO& io = ImGui::GetIO();
175  if (io.ConfigFlags & ImGuiConfigFlags_NoMouseCursorChange)
176  return false;
177 
178  ImGuiMouseCursor imgui_cursor = ImGui::GetMouseCursor();
179  if (imgui_cursor == ImGuiMouseCursor_None || io.MouseDrawCursor)
180  {
181  // Hide OS mouse cursor if imgui is drawing it or if it wants no cursor
182  ::SetCursor(NULL);
183  }
184  else
185  {
186  // Show OS mouse cursor
187  LPTSTR win32_cursor = IDC_ARROW;
188  switch (imgui_cursor)
189  {
190  case ImGuiMouseCursor_Arrow: win32_cursor = IDC_ARROW; break;
191  case ImGuiMouseCursor_TextInput: win32_cursor = IDC_IBEAM; break;
192  case ImGuiMouseCursor_ResizeAll: win32_cursor = IDC_SIZEALL; break;
193  case ImGuiMouseCursor_ResizeEW: win32_cursor = IDC_SIZEWE; break;
194  case ImGuiMouseCursor_ResizeNS: win32_cursor = IDC_SIZENS; break;
195  case ImGuiMouseCursor_ResizeNESW: win32_cursor = IDC_SIZENESW; break;
196  case ImGuiMouseCursor_ResizeNWSE: win32_cursor = IDC_SIZENWSE; break;
197  case ImGuiMouseCursor_Hand: win32_cursor = IDC_HAND; break;
198  case ImGuiMouseCursor_NotAllowed: win32_cursor = IDC_NO; break;
199  }
200  ::SetCursor(::LoadCursor(NULL, win32_cursor));
201  }
202  return true;
203 }

205 // This code supports multi-viewports (multiple OS Windows mapped into different Dear ImGui viewports)
206 // Because of that, it is a little more complicated than your typical single-viewport binding code!
207 static void ImGui_ImplWin32_UpdateMousePos()
208 {
209  ImGuiIO& io = ImGui::GetIO();
210 
211  // Set OS mouse position if requested (rarely used, only when ImGuiConfigFlags_NavEnableSetMousePos is enabled by user)
212  // (When multi-viewports are enabled, all imgui positions are same as OS positions)
213  if (io.WantSetMousePos)
214  {
215  POINT pos = { (int)io.MousePos.x, (int)io.MousePos.y };
216  if ((io.ConfigFlags & ImGuiConfigFlags_ViewportsEnable) != 0 || ::ClientToScreen(g_hWnd, &pos))
217  ::ClientToScreen(g_hWnd, &pos);
218  ::SetCursorPos(pos.x, pos.y);
219  }
220 
221  io.MousePos = ImVec2(-FLT_MAX, -FLT_MAX);
222  io.MouseHoveredViewport = 0;
223 
224  // Set imgui mouse position
225  POINT mouse_screen_pos;
226  if (!::GetCursorPos(&mouse_screen_pos))
227  return;
228  if (HWND focused_hwnd = ::GetForegroundWindow())
229  {
230  if (::IsChild(focused_hwnd, g_hWnd))
231  focused_hwnd = g_hWnd;
232  if (io.ConfigFlags & ImGuiConfigFlags_ViewportsEnable)
233  {
234  // Multi-viewport mode: mouse position in OS absolute coordinates (io.MousePos is (0,0) when the mouse is on the upper-left of the primary monitor)
235  // This is the position you can get with GetCursorPos(). In theory adding viewport->Pos is also the reverse operation of doing ScreenToClient().
236  if (ImGui::FindViewportByPlatformHandle((void*)focused_hwnd) != NULL)
237  io.MousePos = ImVec2((float)mouse_screen_pos.x, (float)mouse_screen_pos.y);
238  }
239  else
240  {
241  // Single viewport mode: mouse position in client window coordinates (io.MousePos is (0,0) when the mouse is on the upper-left corner of the app window.)
242  // This is the position you can get with GetCursorPos() + ScreenToClient() or from WM_MOUSEMOVE.
243  if (focused_hwnd == g_hWnd)
244  {
245  POINT mouse_client_pos = mouse_screen_pos;
246  ::ScreenToClient(focused_hwnd, &mouse_client_pos);
247  io.MousePos = ImVec2((float)mouse_client_pos.x, (float)mouse_client_pos.y);
248  }
249  }
250  }
251 
252  // (Optional) When using multiple viewports: set io.MouseHoveredViewport to the viewport the OS mouse cursor is hovering.
253  // Important: this information is not easy to provide and many high-level windowing library won't be able to provide it correctly, because
254  // - This is _ignoring_ viewports with the ImGuiViewportFlags_NoInputs flag (pass-through windows).
255  // - This is _regardless_ of whether another viewport is focused or being dragged from.
256  // If ImGuiBackendFlags_HasMouseHoveredViewport is not set by the backend, imgui will ignore this field and infer the information by relying on the
257  // rectangles and last focused time of every viewports it knows about. It will be unaware of foreign windows that may be sitting between or over your windows.
258  if (HWND hovered_hwnd = ::WindowFromPoint(mouse_screen_pos))
259  if (ImGuiViewport* viewport = ImGui::FindViewportByPlatformHandle((void*)hovered_hwnd))
260  if ((viewport->Flags & ImGuiViewportFlags_NoInputs) == 0) // FIXME: We still get our NoInputs window with WM_NCHITTEST/HTTRANSPARENT code when decorated?
261  io.MouseHoveredViewport = viewport->ID;
262 }

264 // Gamepad navigation mapping
265 static void ImGui_ImplWin32_UpdateGamepads()
266 {
267 #ifndef IMGUI_IMPL_WIN32_DISABLE_GAMEPAD
268  ImGuiIO& io = ImGui::GetIO();
269  memset(io.NavInputs, 0, sizeof(io.NavInputs));
270  if ((io.ConfigFlags & ImGuiConfigFlags_NavEnableGamepad) == 0)
271  return;
272 
273  // Calling XInputGetState() every frame on disconnected gamepads is unfortunately too slow.
274  // Instead we refresh gamepad availability by calling XInputGetCapabilities() _only_ after receiving WM_DEVICECHANGE.
275  if (g_WantUpdateHasGamepad)
276  {
277  XINPUT_CAPABILITIES caps;
278  g_HasGamepad = g_XInputGetCapabilities ? (g_XInputGetCapabilities(0, XINPUT_FLAG_GAMEPAD, &caps) == ERROR_SUCCESS) : false;
279  g_WantUpdateHasGamepad = false;
280  }
281 
282  io.BackendFlags &= ~ImGuiBackendFlags_HasGamepad;
283  XINPUT_STATE xinput_state;
284  if (g_HasGamepad && g_XInputGetState && g_XInputGetState(0, &xinput_state) == ERROR_SUCCESS)
285  {
286  const XINPUT_GAMEPAD& gamepad = xinput_state.Gamepad;
287  io.BackendFlags |= ImGuiBackendFlags_HasGamepad;
288 
289  #define MAP_BUTTON(NAV_NO, BUTTON_ENUM) { io.NavInputs[NAV_NO] = (gamepad.wButtons & BUTTON_ENUM) ? 1.0f : 0.0f; }
290  #define MAP_ANALOG(NAV_NO, VALUE, V0, V1) { float vn = (float)(VALUE - V0) / (float)(V1 - V0); if (vn > 1.0f) vn = 1.0f; if (vn > 0.0f && io.NavInputs[NAV_NO] < vn) io.NavInputs[NAV_NO] = vn; }
291  MAP_BUTTON(ImGuiNavInput_Activate, XINPUT_GAMEPAD_A); // Cross / A
292  MAP_BUTTON(ImGuiNavInput_Cancel, XINPUT_GAMEPAD_B); // Circle / B
293  MAP_BUTTON(ImGuiNavInput_Menu, XINPUT_GAMEPAD_X); // Square / X
294  MAP_BUTTON(ImGuiNavInput_Input, XINPUT_GAMEPAD_Y); // Triangle / Y
295  MAP_BUTTON(ImGuiNavInput_DpadLeft, XINPUT_GAMEPAD_DPAD_LEFT); // D-Pad Left
296  MAP_BUTTON(ImGuiNavInput_DpadRight, XINPUT_GAMEPAD_DPAD_RIGHT); // D-Pad Right
297  MAP_BUTTON(ImGuiNavInput_DpadUp, XINPUT_GAMEPAD_DPAD_UP); // D-Pad Up
298  MAP_BUTTON(ImGuiNavInput_DpadDown, XINPUT_GAMEPAD_DPAD_DOWN); // D-Pad Down
299  MAP_BUTTON(ImGuiNavInput_FocusPrev, XINPUT_GAMEPAD_LEFT_SHOULDER); // L1 / LB
300  MAP_BUTTON(ImGuiNavInput_FocusNext, XINPUT_GAMEPAD_RIGHT_SHOULDER); // R1 / RB
301  MAP_BUTTON(ImGuiNavInput_TweakSlow, XINPUT_GAMEPAD_LEFT_SHOULDER); // L1 / LB
302  MAP_BUTTON(ImGuiNavInput_TweakFast, XINPUT_GAMEPAD_RIGHT_SHOULDER); // R1 / RB
303  MAP_ANALOG(ImGuiNavInput_LStickLeft, gamepad.sThumbLX, -XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE, -32768);
304  MAP_ANALOG(ImGuiNavInput_LStickRight, gamepad.sThumbLX, +XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE, +32767);
305  MAP_ANALOG(ImGuiNavInput_LStickUp, gamepad.sThumbLY, +XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE, +32767);
306  MAP_ANALOG(ImGuiNavInput_LStickDown, gamepad.sThumbLY, -XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE, -32767);
307  #undef MAP_BUTTON
308  #undef MAP_ANALOG
309  }
310 #endif // #ifndef IMGUI_IMPL_WIN32_DISABLE_GAMEPAD
311 }

340 void ImGui_ImplWin32_NewFrame()
341 {
342  ImGuiIO& io = ImGui::GetIO();
343  IM_ASSERT(io.Fonts->IsBuilt() && "Font atlas not built! It is generally built by the renderer backend. Missing call to renderer _NewFrame() function? e.g. ImGui_ImplOpenGL3_NewFrame().");
344 
345  // Setup display size (every frame to accommodate for window resizing)
346  RECT rect = { 0, 0, 0, 0 };
347  ::GetClientRect(g_hWnd, &rect);
348  io.DisplaySize = ImVec2((float)(rect.right - rect.left), (float)(rect.bottom - rect.top));
349  if (g_WantUpdateMonitors)
350  ImGui_ImplWin32_UpdateMonitors();
351 
352  // Setup time step
353  INT64 current_time = 0;
354  ::QueryPerformanceCounter((LARGE_INTEGER*)&current_time);
355  io.DeltaTime = (float)(current_time - g_Time) / g_TicksPerSecond;
356  g_Time = current_time;
357 
358  // Read keyboard modifiers inputs
359  io.KeyCtrl = (::GetKeyState(VK_CONTROL) & 0x8000) != 0;
360  io.KeyShift = (::GetKeyState(VK_SHIFT) & 0x8000) != 0;
361  io.KeyAlt = (::GetKeyState(VK_MENU) & 0x8000) != 0;
362  io.KeySuper = false;
363  // io.KeysDown[], io.MousePos, io.MouseDown[], io.MouseWheel: filled by the WndProc handler below.
364 
365  // Update OS mouse position
366  ImGui_ImplWin32_UpdateMousePos();
367 
368  // Update OS mouse cursor with the cursor requested by imgui
369  ImGuiMouseCursor mouse_cursor = io.MouseDrawCursor ? ImGuiMouseCursor_None : ImGui::GetMouseCursor();
370  if (g_LastMouseCursor != mouse_cursor)
371  {
372  g_LastMouseCursor = mouse_cursor;
373  ImGui_ImplWin32_UpdateMouseCursor();
374  }
375 
376  // Update game controllers (if enabled and available)
377  ImGui_ImplWin32_UpdateGamepads();
378 }

400 IMGUI_IMPL_API LRESULT ImGui_ImplWin32_WndProcHandler(HWND hwnd, UINT msg, WPARAM wParam, LPARAM lParam)
401 {
402  if (ImGui::GetCurrentContext() == NULL)
403  return 0;
404 
405  ImGuiIO& io = ImGui::GetIO();
406  switch (msg)
407  {
408  case WM_LBUTTONDOWN: case WM_LBUTTONDBLCLK:
409  case WM_RBUTTONDOWN: case WM_RBUTTONDBLCLK:
410  case WM_MBUTTONDOWN: case WM_MBUTTONDBLCLK:
411  case WM_XBUTTONDOWN: case WM_XBUTTONDBLCLK:
412  {
413  int button = 0;
414  if (msg == WM_LBUTTONDOWN || msg == WM_LBUTTONDBLCLK) { button = 0; }
415  if (msg == WM_RBUTTONDOWN || msg == WM_RBUTTONDBLCLK) { button = 1; }
416  if (msg == WM_MBUTTONDOWN || msg == WM_MBUTTONDBLCLK) { button = 2; }
417  if (msg == WM_XBUTTONDOWN || msg == WM_XBUTTONDBLCLK) { button = (GET_XBUTTON_WPARAM(wParam) == XBUTTON1) ? 3 : 4; }
418  if (!ImGui::IsAnyMouseDown() && ::GetCapture() == NULL)
419  ::SetCapture(hwnd);
420  io.MouseDown[button] = true;
421  return 0;
422  }
423  case WM_LBUTTONUP:
424  case WM_RBUTTONUP:
425  case WM_MBUTTONUP:
426  case WM_XBUTTONUP:
427  {
428  int button = 0;
429  if (msg == WM_LBUTTONUP) { button = 0; }
430  if (msg == WM_RBUTTONUP) { button = 1; }
431  if (msg == WM_MBUTTONUP) { button = 2; }
432  if (msg == WM_XBUTTONUP) { button = (GET_XBUTTON_WPARAM(wParam) == XBUTTON1) ? 3 : 4; }
433  io.MouseDown[button] = false;
434  if (!ImGui::IsAnyMouseDown() && ::GetCapture() == hwnd)
435  ::ReleaseCapture();
436  return 0;
437  }
438  case WM_MOUSEWHEEL:
439  io.MouseWheel += (float)GET_WHEEL_DELTA_WPARAM(wParam) / (float)WHEEL_DELTA;
440  return 0;
441  case WM_MOUSEHWHEEL:
442  io.MouseWheelH += (float)GET_WHEEL_DELTA_WPARAM(wParam) / (float)WHEEL_DELTA;
443  return 0;
444  case WM_KEYDOWN:
445  case WM_SYSKEYDOWN:
446  if (wParam < 256)
447  io.KeysDown[wParam] = 1;
448  return 0;
449  case WM_KEYUP:
450  case WM_SYSKEYUP:
451  if (wParam < 256)
452  io.KeysDown[wParam] = 0;
453  return 0;
454  case WM_CHAR:
455  // You can also use ToAscii()+GetKeyboardState() to retrieve characters.
456  if (wParam > 0 && wParam < 0x10000)
457  io.AddInputCharacterUTF16((unsigned short)wParam);
458  return 0;
459  case WM_SETCURSOR:
460  if (LOWORD(lParam) == HTCLIENT && ImGui_ImplWin32_UpdateMouseCursor())
461  return 1;
462  return 0;
463  case WM_DEVICECHANGE:
464  if ((UINT)wParam == DBT_DEVNODES_CHANGED)
465  g_WantUpdateHasGamepad = true;
466  return 0;
467  case WM_DISPLAYCHANGE:
468  g_WantUpdateMonitors = true;
469  return 0;
470  }
471  return 0;
472 }
