61 struct nk_vec2 cursor_new, cursor_pos, cursor_vel;
62 enum { cursor_sync_query, cursor_input_message } cursor_method = cursor_sync_query;

64 void sample_input(GLFWwindow* window)
65 {
66  float a = .25; // exponential smoothing factor
67 
68  if (cursor_method == cursor_sync_query) {
69  double x, y;
70  glfwGetCursorPos(window, &x, &y);
71  cursor_new.x = (float) x;
72  cursor_new.y = (float) y;
73  }
74 
75  cursor_vel.x = (cursor_new.x - cursor_pos.x) * a + cursor_vel.x * (1 - a);
76  cursor_vel.y = (cursor_new.y - cursor_pos.y) * a + cursor_vel.y * (1 - a);
77  cursor_pos = cursor_new;
78 }

99 void swap_buffers(GLFWwindow* window)
100 {
101  glfwSwapBuffers(window);
102 
103  if (swap_clear)
104  glClear(GL_COLOR_BUFFER_BIT);
105 
106  if (swap_finish)
107  glFinish();
108 
109  if (swap_occlusion_query) {
110  GLint occlusion_result;
111  if (!occlusion_query)
112  glGenQueries(1, &occlusion_query);
113  glBeginQuery(GL_SAMPLES_PASSED, occlusion_query);
114  glBegin(GL_POINTS);
115  glVertex2f(0, 0);
116  glEnd();
117  glEndQuery(GL_SAMPLES_PASSED);
118  glGetQueryObjectiv(occlusion_query, GL_QUERY_RESULT, &occlusion_result);
119  }
120 
121  if (swap_read_pixels) {
122  unsigned char rgba[4];
123  glReadPixels(0, 0, 1, 1, GL_RGBA, GL_UNSIGNED_BYTE, rgba);
124  }
125 }
