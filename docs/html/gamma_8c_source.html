63 static void chart_ramp_array(struct nk_context* nk,
64  struct nk_color color,

66 {

87 int main(int argc, char** argv)

105 
106  window = glfwCreateWindow(800, 400, "Gamma Test", NULL, NULL);
107  if (!window)
108  {
109  glfwTerminate();
110  exit(EXIT_FAILURE);
111  }
112 
113  {
114  const GLFWgammaramp* ramp = glfwGetGammaRamp(monitor);
115  const size_t array_size = ramp->size * sizeof(short);
116  orig_ramp.size = ramp->size;
117  orig_ramp.red = malloc(array_size);
118  orig_ramp.green = malloc(array_size);
119  orig_ramp.blue = malloc(array_size);
120  memcpy(orig_ramp.red, ramp->red, array_size);
121  memcpy(orig_ramp.green, ramp->green, array_size);
122  memcpy(orig_ramp.blue, ramp->blue, array_size);
123  }
124

127  glfwSwapInterval(1);
128 
129  nk = nk_glfw3_init(window, NK_GLFW3_INSTALL_CALLBACKS);
130  nk_glfw3_font_stash_begin(&atlas);
131  nk_glfw3_font_stash_end();
132 
133  glfwSetKeyCallback(window, key_callback);
134 
135  while (!glfwWindowShouldClose(window))
136  {
137  int width, height;
138  struct nk_rect area;
139 
140  glfwGetWindowSize(window, &width, &height);
141  area = nk_rect(0.f, 0.f, (float) width, (float) height);
142  nk_window_set_bounds(nk, "", area);
143 
144  glClear(GL_COLOR_BUFFER_BIT);

146  if (nk_begin(nk, "", area, 0))
147  {
148  const GLFWgammaramp* ramp;
149 
150  nk_layout_row_dynamic(nk, 30, 3);
151  if (nk_slider_float(nk, 0.1f, &gamma_value, 5.f, 0.1f))
152  glfwSetGamma(monitor, gamma_value);
153  nk_labelf(nk, NK_TEXT_LEFT, "%0.1f", gamma_value);
154  if (nk_button_label(nk, "Revert"))
155  glfwSetGammaRamp(monitor, &orig_ramp);
156 
157  ramp = glfwGetGammaRamp(monitor);
158 
159  nk_layout_row_dynamic(nk, height - 60.f, 3);
160  chart_ramp_array(nk, nk_rgb(255, 0, 0), ramp->size, ramp->red);
161  chart_ramp_array(nk, nk_rgb(0, 255, 0), ramp->size, ramp->green);
162  chart_ramp_array(nk, nk_rgb(0, 0, 255), ramp->size, ramp->blue);
163  }
164 
165  nk_end(nk);
166  nk_glfw3_render(NK_ANTI_ALIASING_ON);
167 
168  glfwSwapBuffers(window);
169  glfwWaitEventsTimeout(1.0);
170  }
171 
172  free(orig_ramp.red);
173  free(orig_ramp.green);
174  free(orig_ramp.blue);
175

177  glfwTerminate();
178  exit(EXIT_SUCCESS);
179 }
