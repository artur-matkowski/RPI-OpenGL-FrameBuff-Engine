162 static void ImGui_ImplDX12_SetupRenderState(ImDrawData* draw_data, ID3D12GraphicsCommandList* ctx, ImGui_ImplDX12_RenderBuffers* fr)
163 {
164  // Setup orthographic projection matrix into our constant buffer
165  // Our visible imgui space lies from draw_data->DisplayPos (top left) to draw_data->DisplayPos+data_data->DisplaySize (bottom right).
166  VERTEX_CONSTANT_BUFFER vertex_constant_buffer;
167  {
168  float L = draw_data->DisplayPos.x;
169  float R = draw_data->DisplayPos.x + draw_data->DisplaySize.x;
170  float T = draw_data->DisplayPos.y;
171  float B = draw_data->DisplayPos.y + draw_data->DisplaySize.y;
172  float mvp[4][4] =
173  {
174  { 2.0f/(R-L), 0.0f, 0.0f, 0.0f },
175  { 0.0f, 2.0f/(T-B), 0.0f, 0.0f },
176  { 0.0f, 0.0f, 0.5f, 0.0f },
177  { (R+L)/(L-R), (T+B)/(B-T), 0.5f, 1.0f },
178  };
179  memcpy(&vertex_constant_buffer.mvp, mvp, sizeof(mvp));
180  }
181 
182  // Setup viewport
183  D3D12_VIEWPORT vp;
184  memset(&vp, 0, sizeof(D3D12_VIEWPORT));
185  vp.Width = draw_data->DisplaySize.x;
186  vp.Height = draw_data->DisplaySize.y;
187  vp.MinDepth = 0.0f;
188  vp.MaxDepth = 1.0f;
189  vp.TopLeftX = vp.TopLeftY = 0.0f;
190  ctx->RSSetViewports(1, &vp);
191 
192  // Bind shader and vertex buffers
193  unsigned int stride = sizeof(ImDrawVert);
194  unsigned int offset = 0;
195  D3D12_VERTEX_BUFFER_VIEW vbv;
196  memset(&vbv, 0, sizeof(D3D12_VERTEX_BUFFER_VIEW));
197  vbv.BufferLocation = fr->VertexBuffer->GetGPUVirtualAddress() + offset;
198  vbv.SizeInBytes = fr->VertexBufferSize * stride;
199  vbv.StrideInBytes = stride;
200  ctx->IASetVertexBuffers(0, 1, &vbv);
201  D3D12_INDEX_BUFFER_VIEW ibv;
202  memset(&ibv, 0, sizeof(D3D12_INDEX_BUFFER_VIEW));
203  ibv.BufferLocation = fr->IndexBuffer->GetGPUVirtualAddress();
204  ibv.SizeInBytes = fr->IndexBufferSize * sizeof(ImDrawIdx);
205  ibv.Format = sizeof(ImDrawIdx) == 2 ? DXGI_FORMAT_R16_UINT : DXGI_FORMAT_R32_UINT;
206  ctx->IASetIndexBuffer(&ibv);
207  ctx->IASetPrimitiveTopology(D3D_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
208  ctx->SetPipelineState(g_pPipelineState);
209  ctx->SetGraphicsRootSignature(g_pRootSignature);
210  ctx->SetGraphicsRoot32BitConstants(0, 16, &vertex_constant_buffer, 0);
211 
212  // Setup blend factor
213  const float blend_factor[4] = { 0.f, 0.f, 0.f, 0.f };
214  ctx->OMSetBlendFactor(blend_factor);
215 }

218 void ImGui_ImplDX12_RenderDrawData(ImDrawData* draw_data, ID3D12GraphicsCommandList* ctx)
219 {
220  // Avoid rendering when minimized
221  if (draw_data->DisplaySize.x <= 0.0f || draw_data->DisplaySize.y <= 0.0f)
222  return;
223 
224  ImGuiViewportDataDx12* render_data = (ImGuiViewportDataDx12*)draw_data->OwnerViewport->RendererUserData;
225  render_data->FrameIndex++;
226  ImGui_ImplDX12_RenderBuffers* fr = &render_data->FrameRenderBuffers[render_data->FrameIndex % g_numFramesInFlight];
227 
228  // Create and grow vertex/index buffers if needed
229  if (fr->VertexBuffer == NULL || fr->VertexBufferSize < draw_data->TotalVtxCount)
230  {
231  SafeRelease(fr->VertexBuffer);
232  fr->VertexBufferSize = draw_data->TotalVtxCount + 5000;
233  D3D12_HEAP_PROPERTIES props;
234  memset(&props, 0, sizeof(D3D12_HEAP_PROPERTIES));
235  props.Type = D3D12_HEAP_TYPE_UPLOAD;
236  props.CPUPageProperty = D3D12_CPU_PAGE_PROPERTY_UNKNOWN;
237  props.MemoryPoolPreference = D3D12_MEMORY_POOL_UNKNOWN;
238  D3D12_RESOURCE_DESC desc;
239  memset(&desc, 0, sizeof(D3D12_RESOURCE_DESC));
240  desc.Dimension = D3D12_RESOURCE_DIMENSION_BUFFER;
241  desc.Width = fr->VertexBufferSize * sizeof(ImDrawVert);
242  desc.Height = 1;
243  desc.DepthOrArraySize = 1;
244  desc.MipLevels = 1;
245  desc.Format = DXGI_FORMAT_UNKNOWN;
246  desc.SampleDesc.Count = 1;
247  desc.Layout = D3D12_TEXTURE_LAYOUT_ROW_MAJOR;
248  desc.Flags = D3D12_RESOURCE_FLAG_NONE;
249  if (g_pd3dDevice->CreateCommittedResource(&props, D3D12_HEAP_FLAG_NONE, &desc, D3D12_RESOURCE_STATE_GENERIC_READ, NULL, IID_PPV_ARGS(&fr->VertexBuffer)) < 0)
250  return;
251  }
252  if (fr->IndexBuffer == NULL || fr->IndexBufferSize < draw_data->TotalIdxCount)
253  {
254  SafeRelease(fr->IndexBuffer);
255  fr->IndexBufferSize = draw_data->TotalIdxCount + 10000;
256  D3D12_HEAP_PROPERTIES props;
257  memset(&props, 0, sizeof(D3D12_HEAP_PROPERTIES));
258  props.Type = D3D12_HEAP_TYPE_UPLOAD;
259  props.CPUPageProperty = D3D12_CPU_PAGE_PROPERTY_UNKNOWN;
260  props.MemoryPoolPreference = D3D12_MEMORY_POOL_UNKNOWN;
261  D3D12_RESOURCE_DESC desc;
262  memset(&desc, 0, sizeof(D3D12_RESOURCE_DESC));
263  desc.Dimension = D3D12_RESOURCE_DIMENSION_BUFFER;
264  desc.Width = fr->IndexBufferSize * sizeof(ImDrawIdx);
265  desc.Height = 1;
266  desc.DepthOrArraySize = 1;
267  desc.MipLevels = 1;
268  desc.Format = DXGI_FORMAT_UNKNOWN;
269  desc.SampleDesc.Count = 1;
270  desc.Layout = D3D12_TEXTURE_LAYOUT_ROW_MAJOR;
271  desc.Flags = D3D12_RESOURCE_FLAG_NONE;
272  if (g_pd3dDevice->CreateCommittedResource(&props, D3D12_HEAP_FLAG_NONE, &desc, D3D12_RESOURCE_STATE_GENERIC_READ, NULL, IID_PPV_ARGS(&fr->IndexBuffer)) < 0)
273  return;
274  }
275 
276  // Upload vertex/index data into a single contiguous GPU buffer
277  void* vtx_resource, *idx_resource;
278  D3D12_RANGE range;
279  memset(&range, 0, sizeof(D3D12_RANGE));
280  if (fr->VertexBuffer->Map(0, &range, &vtx_resource) != S_OK)
281  return;
282  if (fr->IndexBuffer->Map(0, &range, &idx_resource) != S_OK)
283  return;
284  ImDrawVert* vtx_dst = (ImDrawVert*)vtx_resource;
285  ImDrawIdx* idx_dst = (ImDrawIdx*)idx_resource;
286  for (int n = 0; n < draw_data->CmdListsCount; n++)
287  {
288  const ImDrawList* cmd_list = draw_data->CmdLists[n];
289  memcpy(vtx_dst, cmd_list->VtxBuffer.Data, cmd_list->VtxBuffer.Size * sizeof(ImDrawVert));
290  memcpy(idx_dst, cmd_list->IdxBuffer.Data, cmd_list->IdxBuffer.Size * sizeof(ImDrawIdx));
291  vtx_dst += cmd_list->VtxBuffer.Size;
292  idx_dst += cmd_list->IdxBuffer.Size;
293  }
294  fr->VertexBuffer->Unmap(0, &range);
295  fr->IndexBuffer->Unmap(0, &range);
296 
297  // Setup desired DX state
298  ImGui_ImplDX12_SetupRenderState(draw_data, ctx, fr);
299 
300  // Render command lists
301  // (Because we merged all buffers into a single one, we maintain our own offset into them)
302  int global_vtx_offset = 0;
303  int global_idx_offset = 0;
304  ImVec2 clip_off = draw_data->DisplayPos;
305  for (int n = 0; n < draw_data->CmdListsCount; n++)
306  {
307  const ImDrawList* cmd_list = draw_data->CmdLists[n];
308  for (int cmd_i = 0; cmd_i < cmd_list->CmdBuffer.Size; cmd_i++)
309  {
310  const ImDrawCmd* pcmd = &cmd_list->CmdBuffer[cmd_i];
311  if (pcmd->UserCallback != NULL)
312  {
313  // User callback, registered via ImDrawList::AddCallback()
314  // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
315  if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
316  ImGui_ImplDX12_SetupRenderState(draw_data, ctx, fr);
317  else
318  pcmd->UserCallback(cmd_list, pcmd);
319  }
320  else
321  {
322  // Apply Scissor, Bind texture, Draw
323  const D3D12_RECT r = { (LONG)(pcmd->ClipRect.x - clip_off.x), (LONG)(pcmd->ClipRect.y - clip_off.y), (LONG)(pcmd->ClipRect.z - clip_off.x), (LONG)(pcmd->ClipRect.w - clip_off.y) };
324  if (r.right > r.left && r.bottom > r.top)
325  {
326  ctx->SetGraphicsRootDescriptorTable(1, *(D3D12_GPU_DESCRIPTOR_HANDLE*)&pcmd->TextureId);
327  ctx->RSSetScissorRects(1, &r);
328  ctx->DrawIndexedInstanced(pcmd->ElemCount, 1, pcmd->IdxOffset + global_idx_offset, pcmd->VtxOffset + global_vtx_offset, 0);
329  }
330  }
331  }
332  global_idx_offset += cmd_list->IdxBuffer.Size;
333  global_vtx_offset += cmd_list->VtxBuffer.Size;
334  }
335 }

337 static void ImGui_ImplDX12_CreateFontsTexture()
338 {
339  // Build texture atlas
340  ImGuiIO& io = ImGui::GetIO();
341  unsigned char* pixels;
342  int width, height;
343  io.Fonts->GetTexDataAsRGBA32(&pixels, &width, &height);
344 
345  // Upload texture to graphics system
346  {
347  D3D12_HEAP_PROPERTIES props;
348  memset(&props, 0, sizeof(D3D12_HEAP_PROPERTIES));
349  props.Type = D3D12_HEAP_TYPE_DEFAULT;
350  props.CPUPageProperty = D3D12_CPU_PAGE_PROPERTY_UNKNOWN;
351  props.MemoryPoolPreference = D3D12_MEMORY_POOL_UNKNOWN;
352 
353  D3D12_RESOURCE_DESC desc;
354  ZeroMemory(&desc, sizeof(desc));
355  desc.Dimension = D3D12_RESOURCE_DIMENSION_TEXTURE2D;
356  desc.Alignment = 0;
357  desc.Width = width;
358  desc.Height = height;
359  desc.DepthOrArraySize = 1;
360  desc.MipLevels = 1;
361  desc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
362  desc.SampleDesc.Count = 1;
363  desc.SampleDesc.Quality = 0;
364  desc.Layout = D3D12_TEXTURE_LAYOUT_UNKNOWN;
365  desc.Flags = D3D12_RESOURCE_FLAG_NONE;
366 
367  ID3D12Resource* pTexture = NULL;
368  g_pd3dDevice->CreateCommittedResource(&props, D3D12_HEAP_FLAG_NONE, &desc,
369  D3D12_RESOURCE_STATE_COPY_DEST, NULL, IID_PPV_ARGS(&pTexture));
370 
371  UINT uploadPitch = (width * 4 + D3D12_TEXTURE_DATA_PITCH_ALIGNMENT - 1u) & ~(D3D12_TEXTURE_DATA_PITCH_ALIGNMENT - 1u);
372  UINT uploadSize = height * uploadPitch;
373  desc.Dimension = D3D12_RESOURCE_DIMENSION_BUFFER;
374  desc.Alignment = 0;
375  desc.Width = uploadSize;
376  desc.Height = 1;
377  desc.DepthOrArraySize = 1;
378  desc.MipLevels = 1;
379  desc.Format = DXGI_FORMAT_UNKNOWN;
380  desc.SampleDesc.Count = 1;
381  desc.SampleDesc.Quality = 0;
382  desc.Layout = D3D12_TEXTURE_LAYOUT_ROW_MAJOR;
383  desc.Flags = D3D12_RESOURCE_FLAG_NONE;
384 
385  props.Type = D3D12_HEAP_TYPE_UPLOAD;
386  props.CPUPageProperty = D3D12_CPU_PAGE_PROPERTY_UNKNOWN;
387  props.MemoryPoolPreference = D3D12_MEMORY_POOL_UNKNOWN;
388 
389  ID3D12Resource* uploadBuffer = NULL;
390  HRESULT hr = g_pd3dDevice->CreateCommittedResource(&props, D3D12_HEAP_FLAG_NONE, &desc,
391  D3D12_RESOURCE_STATE_GENERIC_READ, NULL, IID_PPV_ARGS(&uploadBuffer));
392  IM_ASSERT(SUCCEEDED(hr));
393 
394  void* mapped = NULL;
395  D3D12_RANGE range = { 0, uploadSize };
396  hr = uploadBuffer->Map(0, &range, &mapped);
397  IM_ASSERT(SUCCEEDED(hr));
398  for (int y = 0; y < height; y++)
399  memcpy((void*) ((uintptr_t) mapped + y * uploadPitch), pixels + y * width * 4, width * 4);
400  uploadBuffer->Unmap(0, &range);
401 
402  D3D12_TEXTURE_COPY_LOCATION srcLocation = {};
403  srcLocation.pResource = uploadBuffer;
404  srcLocation.Type = D3D12_TEXTURE_COPY_TYPE_PLACED_FOOTPRINT;
405  srcLocation.PlacedFootprint.Footprint.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
406  srcLocation.PlacedFootprint.Footprint.Width = width;
407  srcLocation.PlacedFootprint.Footprint.Height = height;
408  srcLocation.PlacedFootprint.Footprint.Depth = 1;
409  srcLocation.PlacedFootprint.Footprint.RowPitch = uploadPitch;
410 
411  D3D12_TEXTURE_COPY_LOCATION dstLocation = {};
412  dstLocation.pResource = pTexture;
413  dstLocation.Type = D3D12_TEXTURE_COPY_TYPE_SUBRESOURCE_INDEX;
414  dstLocation.SubresourceIndex = 0;
415 
416  D3D12_RESOURCE_BARRIER barrier = {};
417  barrier.Type = D3D12_RESOURCE_BARRIER_TYPE_TRANSITION;
418  barrier.Flags = D3D12_RESOURCE_BARRIER_FLAG_NONE;
419  barrier.Transition.pResource = pTexture;
420  barrier.Transition.Subresource = D3D12_RESOURCE_BARRIER_ALL_SUBRESOURCES;
421  barrier.Transition.StateBefore = D3D12_RESOURCE_STATE_COPY_DEST;
422  barrier.Transition.StateAfter = D3D12_RESOURCE_STATE_PIXEL_SHADER_RESOURCE;
423 
424  ID3D12Fence* fence = NULL;
425  hr = g_pd3dDevice->CreateFence(0, D3D12_FENCE_FLAG_NONE, IID_PPV_ARGS(&fence));
426  IM_ASSERT(SUCCEEDED(hr));
427 
428  HANDLE event = CreateEvent(0, 0, 0, 0);
429  IM_ASSERT(event != NULL);
430 
431  D3D12_COMMAND_QUEUE_DESC queueDesc = {};
432  queueDesc.Type = D3D12_COMMAND_LIST_TYPE_DIRECT;
433  queueDesc.Flags = D3D12_COMMAND_QUEUE_FLAG_NONE;
434  queueDesc.NodeMask = 1;
435 
436  ID3D12CommandQueue* cmdQueue = NULL;
437  hr = g_pd3dDevice->CreateCommandQueue(&queueDesc, IID_PPV_ARGS(&cmdQueue));
438  IM_ASSERT(SUCCEEDED(hr));
439 
440  ID3D12CommandAllocator* cmdAlloc = NULL;
441  hr = g_pd3dDevice->CreateCommandAllocator(D3D12_COMMAND_LIST_TYPE_DIRECT, IID_PPV_ARGS(&cmdAlloc));
442  IM_ASSERT(SUCCEEDED(hr));
443 
444  ID3D12GraphicsCommandList* cmdList = NULL;
445  hr = g_pd3dDevice->CreateCommandList(0, D3D12_COMMAND_LIST_TYPE_DIRECT, cmdAlloc, NULL, IID_PPV_ARGS(&cmdList));
446  IM_ASSERT(SUCCEEDED(hr));
447 
448  cmdList->CopyTextureRegion(&dstLocation, 0, 0, 0, &srcLocation, NULL);
449  cmdList->ResourceBarrier(1, &barrier);
450 
451  hr = cmdList->Close();
452  IM_ASSERT(SUCCEEDED(hr));
453 
454  cmdQueue->ExecuteCommandLists(1, (ID3D12CommandList* const*)&cmdList);
455  hr = cmdQueue->Signal(fence, 1);
456  IM_ASSERT(SUCCEEDED(hr));
457 
458  fence->SetEventOnCompletion(1, event);
459  WaitForSingleObject(event, INFINITE);
460 
461  cmdList->Release();
462  cmdAlloc->Release();
463  cmdQueue->Release();
464  CloseHandle(event);
465  fence->Release();
466  uploadBuffer->Release();
467 
468  // Create texture view
469  D3D12_SHADER_RESOURCE_VIEW_DESC srvDesc;
470  ZeroMemory(&srvDesc, sizeof(srvDesc));
471  srvDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
472  srvDesc.ViewDimension = D3D12_SRV_DIMENSION_TEXTURE2D;
473  srvDesc.Texture2D.MipLevels = desc.MipLevels;
474  srvDesc.Texture2D.MostDetailedMip = 0;
475  srvDesc.Shader4ComponentMapping = D3D12_DEFAULT_SHADER_4_COMPONENT_MAPPING;
476  g_pd3dDevice->CreateShaderResourceView(pTexture, &srvDesc, g_hFontSrvCpuDescHandle);
477  SafeRelease(g_pFontTextureResource);
478  g_pFontTextureResource = pTexture;
479  }

481  // Store our identifier

796 static void ImGui_ImplDX12_CreateWindow(ImGuiViewport* viewport)
797 {
798  ImGuiViewportDataDx12* data = IM_NEW(ImGuiViewportDataDx12)();
799  viewport->RendererUserData = data;
800 
801  // PlatformHandleRaw should always be a HWND, whereas PlatformHandle might be a higher-level handle (e.g. GLFWWindow*, SDL_Window*).
802  // Some backends will leave PlatformHandleRaw NULL, in which case we assume PlatformHandle will contain the HWND.
803  HWND hwnd = viewport->PlatformHandleRaw ? (HWND)viewport->PlatformHandleRaw : (HWND)viewport->PlatformHandle;
804  IM_ASSERT(hwnd != 0);
805 
806  data->FrameIndex = UINT_MAX;
807 
808  // Create command queue.
809  D3D12_COMMAND_QUEUE_DESC queue_desc = {};
810  queue_desc.Flags = D3D12_COMMAND_QUEUE_FLAG_NONE;
811  queue_desc.Type = D3D12_COMMAND_LIST_TYPE_DIRECT;
812 
813  HRESULT res = S_OK;
814  res = g_pd3dDevice->CreateCommandQueue(&queue_desc, IID_PPV_ARGS(&data->CommandQueue));
815  IM_ASSERT(res == S_OK);
816 
817  // Create command allocator.
818  for (UINT i = 0; i < g_numFramesInFlight; ++i)
819  {
820  res = g_pd3dDevice->CreateCommandAllocator(D3D12_COMMAND_LIST_TYPE_DIRECT, IID_PPV_ARGS(&data->FrameCtx[i].CommandAllocator));
821  IM_ASSERT(res == S_OK);
822  }
823 
824  // Create command list.
825  res = g_pd3dDevice->CreateCommandList(0, D3D12_COMMAND_LIST_TYPE_DIRECT, data->FrameCtx[0].CommandAllocator, NULL, IID_PPV_ARGS(&data->CommandList));
826  IM_ASSERT(res == S_OK);
827  data->CommandList->Close();
828 
829  // Create fence.
830  res = g_pd3dDevice->CreateFence(0, D3D12_FENCE_FLAG_NONE, IID_PPV_ARGS(&data->Fence));
831  IM_ASSERT(res == S_OK);
832 
833  data->FenceEvent = CreateEvent(NULL, FALSE, FALSE, NULL);
834  IM_ASSERT(data->FenceEvent != NULL);
835 
836  // Create swap chain
837  // FIXME-VIEWPORT: May want to copy/inherit swap chain settings from the user/application.
838  DXGI_SWAP_CHAIN_DESC1 sd1;
839  ZeroMemory(&sd1, sizeof(sd1));
840  sd1.BufferCount = g_numFramesInFlight;
841  sd1.Width = (UINT)viewport->Size.x;
842  sd1.Height = (UINT)viewport->Size.y;
843  sd1.Format = g_RTVFormat;
844  sd1.BufferUsage = DXGI_USAGE_RENDER_TARGET_OUTPUT;
845  sd1.SampleDesc.Count = 1;
846  sd1.SampleDesc.Quality = 0;
847  sd1.SwapEffect = DXGI_SWAP_EFFECT_FLIP_DISCARD;
848  sd1.AlphaMode = DXGI_ALPHA_MODE_UNSPECIFIED;
849  sd1.Scaling = DXGI_SCALING_STRETCH;
850  sd1.Stereo = FALSE;
851 
852  IDXGIFactory4* dxgi_factory = NULL;
853  res = ::CreateDXGIFactory1(IID_PPV_ARGS(&dxgi_factory));
854  IM_ASSERT(res == S_OK);
855 
856  IDXGISwapChain1* swap_chain = NULL;
857  res = dxgi_factory->CreateSwapChainForHwnd(data->CommandQueue, hwnd, &sd1, NULL, NULL, &swap_chain);
858  IM_ASSERT(res == S_OK);
859 
860  dxgi_factory->Release();
861 
862  // Or swapChain.As(&mSwapChain)
863  IM_ASSERT(data->SwapChain == NULL);
864  swap_chain->QueryInterface(IID_PPV_ARGS(&data->SwapChain));
865  swap_chain->Release();
866 
867  // Create the render targets
868  if (data->SwapChain)
869  {
870  D3D12_DESCRIPTOR_HEAP_DESC desc = {};
871  desc.Type = D3D12_DESCRIPTOR_HEAP_TYPE_RTV;
872  desc.NumDescriptors = g_numFramesInFlight;
873  desc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_NONE;
874  desc.NodeMask = 1;
875 
876  HRESULT hr = g_pd3dDevice->CreateDescriptorHeap(&desc, IID_PPV_ARGS(&data->RtvDescHeap));
877  IM_ASSERT(hr == S_OK);
878 
879  SIZE_T rtv_descriptor_size = g_pd3dDevice->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_RTV);
880  D3D12_CPU_DESCRIPTOR_HANDLE rtv_handle = data->RtvDescHeap->GetCPUDescriptorHandleForHeapStart();
881  for (UINT i = 0; i < g_numFramesInFlight; i++)
882  {
883  data->FrameCtx[i].RenderTargetCpuDescriptors = rtv_handle;
884  rtv_handle.ptr += rtv_descriptor_size;
885  }
886 
887  ID3D12Resource* back_buffer;
888  for (UINT i = 0; i < g_numFramesInFlight; i++)
889  {
890  IM_ASSERT(data->FrameCtx[i].RenderTarget == NULL);
891  data->SwapChain->GetBuffer(i, IID_PPV_ARGS(&back_buffer));
892  g_pd3dDevice->CreateRenderTargetView(back_buffer, NULL, data->FrameCtx[i].RenderTargetCpuDescriptors);
893  data->FrameCtx[i].RenderTarget = back_buffer;
894  }
895  }
896 
897  for (UINT i = 0; i < g_numFramesInFlight; i++)
898  ImGui_ImplDX12_DestroyRenderBuffers(&data->FrameRenderBuffers[i]);
899 }

901 static void ImGui_WaitForPendingOperations(ImGuiViewportDataDx12* data)
902 {
903  HRESULT hr = S_FALSE;
904  if (data && data->CommandQueue && data->Fence && data->FenceEvent)
905  {
906  hr = data->CommandQueue->Signal(data->Fence, ++data->FenceSignaledValue);
907  IM_ASSERT(hr == S_OK);
908  ::WaitForSingleObject(data->FenceEvent, 0); // Reset any forgotten waits
909  hr = data->Fence->SetEventOnCompletion(data->FenceSignaledValue, data->FenceEvent);
910  IM_ASSERT(hr == S_OK);
911  ::WaitForSingleObject(data->FenceEvent, INFINITE);
912  }

963 static void ImGui_ImplDX12_RenderWindow(ImGuiViewport* viewport, void*)
964 {
