251 - (MetalBuffer *)dequeueReusableBufferOfLength:(NSUInteger)length device:(id<MTLDevice>)device
252 {
253  NSTimeInterval now = [NSDate date].timeIntervalSince1970;
254 
255  // Purge old buffers that haven't been useful for a while
256  if (now - self.lastBufferCachePurge > 1.0)
257  {
258  NSMutableArray *survivors = [NSMutableArray array];
259  for (MetalBuffer *candidate in self.bufferCache)
260  {
261  if (candidate.lastReuseTime > self.lastBufferCachePurge)
262  {
263  [survivors addObject:candidate];
264  }
265  }
266  self.bufferCache = [survivors mutableCopy];
267  self.lastBufferCachePurge = now;
268  }
269 
270  // See if we have a buffer we can reuse
271  MetalBuffer *bestCandidate = nil;
272  for (MetalBuffer *candidate in self.bufferCache)
273  if (candidate.buffer.length >= length && (bestCandidate == nil || bestCandidate.lastReuseTime > candidate.lastReuseTime))
274  bestCandidate = candidate;
275 
276  if (bestCandidate != nil)
277  {
278  [self.bufferCache removeObject:bestCandidate];
279  bestCandidate.lastReuseTime = now;
280  return bestCandidate;
281  }
282 
283  // No luck; make a new buffer
284  id<MTLBuffer> backing = [device newBufferWithLength:length options:MTLResourceStorageModeShared];
285  return [[MetalBuffer alloc] initWithBuffer:backing];
286 }

311 {
312  NSError *error = nil;
313 
314  NSString *shaderSource = @""
315  "#include <metal_stdlib>\n"
316  "using namespace metal;\n"
317  "\n"
318  "struct Uniforms {\n"
319  " float4x4 projectionMatrix;\n"
320  "};\n"
321  "\n"
322  "struct VertexIn {\n"
323  " float2 position [[attribute(0)]];\n"
324  " float2 texCoords [[attribute(1)]];\n"
325  " uchar4 color [[attribute(2)]];\n"
326  "};\n"
327  "\n"
328  "struct VertexOut {\n"
329  " float4 position [[position]];\n"
330  " float2 texCoords;\n"
331  " float4 color;\n"
332  "};\n"
333  "\n"
334  "vertex VertexOut vertex_main(VertexIn in [[stage_in]],\n"
335  " constant Uniforms &uniforms [[buffer(1)]]) {\n"
336  " VertexOut out;\n"
337  " out.position = uniforms.projectionMatrix * float4(in.position, 0, 1);\n"
338  " out.texCoords = in.texCoords;\n"
339  " out.color = float4(in.color) / float4(255.0);\n"
340  " return out;\n"
341  "}\n"
342  "\n"
343  "fragment half4 fragment_main(VertexOut in [[stage_in]],\n"
344  " texture2d<half, access::sample> texture [[texture(0)]]) {\n"
345  " constexpr sampler linearSampler(coord::normalized, min_filter::linear, mag_filter::linear, mip_filter::linear);\n"
346  " half4 texColor = texture.sample(linearSampler, in.texCoords);\n"
347  " return half4(in.color) * texColor;\n"
348  "}\n";
349 
350  id<MTLLibrary> library = [device newLibraryWithSource:shaderSource options:nil error:&error];
351  if (library == nil)
352  {
353  NSLog(@"Error: failed to create Metal library: %@", error);
354  return nil;
355  }
356 
357  id<MTLFunction> vertexFunction = [library newFunctionWithName:@"vertex_main"];
358  id<MTLFunction> fragmentFunction = [library newFunctionWithName:@"fragment_main"];
359 
360  if (vertexFunction == nil || fragmentFunction == nil)
361  {
362  NSLog(@"Error: failed to find Metal shader functions in library: %@", error);
363  return nil;
364  }

366  MTLVertexDescriptor *vertexDescriptor = [MTLVertexDescriptor vertexDescriptor];
367  vertexDescriptor.attributes[0].offset = IM_OFFSETOF(ImDrawVert, pos);
368  vertexDescriptor.attributes[0].format = MTLVertexFormatFloat2; // position
369  vertexDescriptor.attributes[0].bufferIndex = 0;
370  vertexDescriptor.attributes[1].offset = IM_OFFSETOF(ImDrawVert, uv);
371  vertexDescriptor.attributes[1].format = MTLVertexFormatFloat2; // texCoords
372  vertexDescriptor.attributes[1].bufferIndex = 0;
373  vertexDescriptor.attributes[2].offset = IM_OFFSETOF(ImDrawVert, col);
374  vertexDescriptor.attributes[2].format = MTLVertexFormatUChar4; // color
375  vertexDescriptor.attributes[2].bufferIndex = 0;
376  vertexDescriptor.layouts[0].stepRate = 1;
377  vertexDescriptor.layouts[0].stepFunction = MTLVertexStepFunctionPerVertex;
378  vertexDescriptor.layouts[0].stride = sizeof(ImDrawVert);
379

381  pipelineDescriptor.vertexFunction = vertexFunction;
382  pipelineDescriptor.fragmentFunction = fragmentFunction;
383  pipelineDescriptor.vertexDescriptor = vertexDescriptor;
384  pipelineDescriptor.sampleCount = self.framebufferDescriptor.sampleCount;
385  pipelineDescriptor.colorAttachments[0].pixelFormat = self.framebufferDescriptor.colorPixelFormat;
386  pipelineDescriptor.colorAttachments[0].blendingEnabled = YES;
387  pipelineDescriptor.colorAttachments[0].rgbBlendOperation = MTLBlendOperationAdd;
388  pipelineDescriptor.colorAttachments[0].alphaBlendOperation = MTLBlendOperationAdd;
389  pipelineDescriptor.colorAttachments[0].sourceRGBBlendFactor = MTLBlendFactorSourceAlpha;
390  pipelineDescriptor.colorAttachments[0].sourceAlphaBlendFactor = MTLBlendFactorSourceAlpha;
391  pipelineDescriptor.colorAttachments[0].destinationRGBBlendFactor = MTLBlendFactorOneMinusSourceAlpha;
392  pipelineDescriptor.colorAttachments[0].destinationAlphaBlendFactor = MTLBlendFactorOneMinusSourceAlpha;
393  pipelineDescriptor.depthAttachmentPixelFormat = self.framebufferDescriptor.depthPixelFormat;
394  pipelineDescriptor.stencilAttachmentPixelFormat = self.framebufferDescriptor.stencilPixelFormat;

410 - (void)setupRenderState:(ImDrawData *)drawData
411  commandBuffer:(id<MTLCommandBuffer>)commandBuffer
412  commandEncoder:(id<MTLRenderCommandEncoder>)commandEncoder
413  renderPipelineState:(id<MTLRenderPipelineState>)renderPipelineState
414  vertexBuffer:(MetalBuffer *)vertexBuffer
415  vertexBufferOffset:(size_t)vertexBufferOffset
416 {
417  [commandEncoder setCullMode:MTLCullModeNone];
418  [commandEncoder setDepthStencilState:g_sharedMetalContext.depthStencilState];
419 
420  // Setup viewport, orthographic projection matrix
421  // Our visible imgui space lies from draw_data->DisplayPos (top left) to
422  // draw_data->DisplayPos+data_data->DisplaySize (bottom right). DisplayMin is typically (0,0) for single viewport apps.
423  MTLViewport viewport =
424  {
425  .originX = 0.0,
426  .originY = 0.0,
427  .width = (double)(drawData->DisplaySize.x * drawData->FramebufferScale.x),
428  .height = (double)(drawData->DisplaySize.y * drawData->FramebufferScale.y),
429  .znear = 0.0,
430  .zfar = 1.0
431  };
432  [commandEncoder setViewport:viewport];
433 
434  float L = drawData->DisplayPos.x;
435  float R = drawData->DisplayPos.x + drawData->DisplaySize.x;
436  float T = drawData->DisplayPos.y;
437  float B = drawData->DisplayPos.y + drawData->DisplaySize.y;
438  float N = (float)viewport.znear;
439  float F = (float)viewport.zfar;
440  const float ortho_projection[4][4] =
441  {
442  { 2.0f/(R-L), 0.0f, 0.0f, 0.0f },
443  { 0.0f, 2.0f/(T-B), 0.0f, 0.0f },
444  { 0.0f, 0.0f, 1/(F-N), 0.0f },
445  { (R+L)/(L-R), (T+B)/(B-T), N/(F-N), 1.0f },
446  };
447  [commandEncoder setVertexBytes:&ortho_projection length:sizeof(ortho_projection) atIndex:1];
448 
449  [commandEncoder setRenderPipelineState:renderPipelineState];
450 
451  [commandEncoder setVertexBuffer:vertexBuffer.buffer offset:0 atIndex:0];
452  [commandEncoder setVertexBufferOffset:vertexBufferOffset atIndex:0];
453 }

455 - (void)renderDrawData:(ImDrawData *)drawData
456  commandBuffer:(id<MTLCommandBuffer>)commandBuffer
457  commandEncoder:(id<MTLRenderCommandEncoder>)commandEncoder
458 {
459  // Avoid rendering when minimized, scale coordinates for retina displays (screen coordinates != framebuffer coordinates)
460  int fb_width = (int)(drawData->DisplaySize.x * drawData->FramebufferScale.x);
461  int fb_height = (int)(drawData->DisplaySize.y * drawData->FramebufferScale.y);
462  if (fb_width <= 0 || fb_height <= 0 || drawData->CmdListsCount == 0)
463  return;
464 
465  id<MTLRenderPipelineState> renderPipelineState = [self renderPipelineStateForFrameAndDevice:commandBuffer.device];
466 
467  size_t vertexBufferLength = (size_t)drawData->TotalVtxCount * sizeof(ImDrawVert);
468  size_t indexBufferLength = (size_t)drawData->TotalIdxCount * sizeof(ImDrawIdx);
469  MetalBuffer* vertexBuffer = [self dequeueReusableBufferOfLength:vertexBufferLength device:commandBuffer.device];
470  MetalBuffer* indexBuffer = [self dequeueReusableBufferOfLength:indexBufferLength device:commandBuffer.device];
471 
472  [self setupRenderState:drawData commandBuffer:commandBuffer commandEncoder:commandEncoder renderPipelineState:renderPipelineState vertexBuffer:vertexBuffer vertexBufferOffset:0];
473 
474  // Will project scissor/clipping rectangles into framebuffer space
475  ImVec2 clip_off = drawData->DisplayPos; // (0,0) unless using multi-viewports
476  ImVec2 clip_scale = drawData->FramebufferScale; // (1,1) unless using retina display which are often (2,2)
477 
478  // Render command lists
479  size_t vertexBufferOffset = 0;
480  size_t indexBufferOffset = 0;
481  for (int n = 0; n < drawData->CmdListsCount; n++)
482  {
483  const ImDrawList* cmd_list = drawData->CmdLists[n];
484 
485  memcpy((char *)vertexBuffer.buffer.contents + vertexBufferOffset, cmd_list->VtxBuffer.Data, (size_t)cmd_list->VtxBuffer.Size * sizeof(ImDrawVert));
486  memcpy((char *)indexBuffer.buffer.contents + indexBufferOffset, cmd_list->IdxBuffer.Data, (size_t)cmd_list->IdxBuffer.Size * sizeof(ImDrawIdx));
487 
488  for (int cmd_i = 0; cmd_i < cmd_list->CmdBuffer.Size; cmd_i++)
489  {
490  const ImDrawCmd* pcmd = &cmd_list->CmdBuffer[cmd_i];
491  if (pcmd->UserCallback)
492  {
493  // User callback, registered via ImDrawList::AddCallback()
494  // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
495  if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
496  [self setupRenderState:drawData commandBuffer:commandBuffer commandEncoder:commandEncoder renderPipelineState:renderPipelineState vertexBuffer:vertexBuffer vertexBufferOffset:vertexBufferOffset];
497  else
498  pcmd->UserCallback(cmd_list, pcmd);
499  }
500  else
501  {
502  // Project scissor/clipping rectangles into framebuffer space
503  ImVec4 clip_rect;
504  clip_rect.x = (pcmd->ClipRect.x - clip_off.x) * clip_scale.x;
505  clip_rect.y = (pcmd->ClipRect.y - clip_off.y) * clip_scale.y;
506  clip_rect.z = (pcmd->ClipRect.z - clip_off.x) * clip_scale.x;
507  clip_rect.w = (pcmd->ClipRect.w - clip_off.y) * clip_scale.y;
508 
509  if (clip_rect.x < fb_width && clip_rect.y < fb_height && clip_rect.z >= 0.0f && clip_rect.w >= 0.0f)
510  {
511  // Apply scissor/clipping rectangle
512  MTLScissorRect scissorRect =
513  {
514  .x = NSUInteger(clip_rect.x),
515  .y = NSUInteger(clip_rect.y),
516  .width = NSUInteger(clip_rect.z - clip_rect.x),
517  .height = NSUInteger(clip_rect.w - clip_rect.y)
518  };
519  [commandEncoder setScissorRect:scissorRect];
520 
521 
522  // Bind texture, Draw
523  if (pcmd->TextureId != NULL)
524  [commandEncoder setFragmentTexture:(__bridge id<MTLTexture>)(pcmd->TextureId) atIndex:0];
525 
526  [commandEncoder setVertexBufferOffset:(vertexBufferOffset + pcmd->VtxOffset * sizeof(ImDrawVert)) atIndex:0];
527  [commandEncoder drawIndexedPrimitives:MTLPrimitiveTypeTriangle
528  indexCount:pcmd->ElemCount
529  indexType:sizeof(ImDrawIdx) == 2 ? MTLIndexTypeUInt16 : MTLIndexTypeUInt32
530  indexBuffer:indexBuffer.buffer
531  indexBufferOffset:indexBufferOffset + pcmd->IdxOffset * sizeof(ImDrawIdx)];
532  }
533  }
534  }
535 
536  vertexBufferOffset += (size_t)cmd_list->VtxBuffer.Size * sizeof(ImDrawVert);
537  indexBufferOffset += (size_t)cmd_list->IdxBuffer.Size * sizeof(ImDrawIdx);
538  }
539 
540  __weak id weakSelf = self;
541  [commandBuffer addCompletedHandler:^(id<MTLCommandBuffer>)
542  {
543  dispatch_async(dispatch_get_main_queue(), ^{
544  [weakSelf enqueueReusableBuffer:vertexBuffer];
545  [weakSelf enqueueReusableBuffer:indexBuffer];
546  });
547  }];
548 }
