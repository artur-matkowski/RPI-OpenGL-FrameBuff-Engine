412 void WaitForLastSubmittedFrame()
413 {
414  FrameContext* frameCtx = &g_frameContext[g_frameIndex % NUM_FRAMES_IN_FLIGHT];
415 
416  UINT64 fenceValue = frameCtx->FenceValue;
417  if (fenceValue == 0)
418  return; // No fence was signaled
419 
420  frameCtx->FenceValue = 0;
421  if (g_fence->GetCompletedValue() >= fenceValue)
422  return;
423 
424  g_fence->SetEventOnCompletion(fenceValue, g_fenceEvent);
425  WaitForSingleObject(g_fenceEvent, INFINITE);
426 }

428 FrameContext* WaitForNextFrameResources()
429 {
430  UINT nextFrameIndex = g_frameIndex + 1;
431  g_frameIndex = nextFrameIndex;
432 
433  HANDLE waitableObjects[] = { g_hSwapChainWaitableObject, NULL };
434  DWORD numWaitableObjects = 1;
435 
436  FrameContext* frameCtx = &g_frameContext[nextFrameIndex % NUM_FRAMES_IN_FLIGHT];
437  UINT64 fenceValue = frameCtx->FenceValue;
438  if (fenceValue != 0) // means no fence was signaled
439  {
440  frameCtx->FenceValue = 0;
441  g_fence->SetEventOnCompletion(fenceValue, g_fenceEvent);
442  waitableObjects[1] = g_fenceEvent;
443  numWaitableObjects = 2;
444  }
445 
446  WaitForMultipleObjects(numWaitableObjects, waitableObjects, TRUE, INFINITE);
447 
448  return frameCtx;
449 }
