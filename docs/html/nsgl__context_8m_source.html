31 // Display link callback for manual swap interval implementation
32 // This is based on a similar workaround added to SDL2
33 //
34 static CVReturn displayLinkCallback(CVDisplayLinkRef displayLink,
35  const CVTimeStamp* now,
36  const CVTimeStamp* outputTime,
37  CVOptionFlags flagsIn,
38  CVOptionFlags* flagsOut,
39  void* userInfo)
40 {
41  _GLFWwindow* window = (_GLFWwindow *) userInfo;
42 
43  const int interval = atomic_load(&window->context.nsgl.swapInterval);
44  if (interval > 0)
45  {
46  [window->context.nsgl.swapIntervalCond lock];
47  window->context.nsgl.swapIntervalsPassed++;
48  [window->context.nsgl.swapIntervalCond signal];
49  [window->context.nsgl.swapIntervalCond unlock];
50  }
51 
52  return kCVReturnSuccess;
53 }

69 static void swapBuffersNSGL(_GLFWwindow* window)
70 {
71  @autoreleasepool {
72 
73  const int interval = atomic_load(&window->context.nsgl.swapInterval);
74  if (interval > 0)
75  {
76  [window->context.nsgl.swapIntervalCond lock];
77  do
78  {
79  [window->context.nsgl.swapIntervalCond wait];
80  } while (window->context.nsgl.swapIntervalsPassed % interval != 0);
81  window->context.nsgl.swapIntervalsPassed = 0;
82  [window->context.nsgl.swapIntervalCond unlock];
83  }
84 
85  // ARP appears to be unnecessary, but this is future-proof
86  [window->context.nsgl.object flushBuffer];
87 
88  } // autoreleasepool
89 }

193  _glfwInputError(GLFW_VERSION_UNAVAILABLE,

199  // Context robustness modes (GL_KHR_robustness) are not yet supported by
200  // macOS but are not a hard constraint, so ignore and continue
201 
202  // Context release behaviors (GL_KHR_context_flush_control) are not yet
203  // supported by macOS but are not a hard constraint, so ignore and continue
204 
205  // Debug contexts (GL_KHR_debug) are not yet supported by macOS but are not

211 #define addAttrib(a) \
212 { \
213  assert((size_t) index < sizeof(attribs) / sizeof(attribs[0])); \
214  attribs[index++] = a; \
215 }
216 #define setAttrib(a, v) { addAttrib(a); addAttrib(v); }
217 
218  NSOpenGLPixelFormatAttribute attribs[40];
219  int index = 0;
220 
221  addAttrib(NSOpenGLPFAAccelerated);
222  addAttrib(NSOpenGLPFAClosestPolicy);
223 
224  if (ctxconfig->nsgl.offline)
225  {
226  addAttrib(NSOpenGLPFAAllowOfflineRenderers);
227  // NOTE: This replaces the NSSupportsAutomaticGraphicsSwitching key in
228  // Info.plist for unbundled applications
229  // HACK: This assumes that NSOpenGLPixelFormat will remain
230  // a straightforward wrapper of its CGL counterpart
231  addAttrib(kCGLPFASupportsAutomaticGraphicsSwitching);
232  }
233 
234 #if MAC_OS_X_VERSION_MAX_ALLOWED >= 101000
235  if (ctxconfig->major >= 4)
236  {
237  setAttrib(NSOpenGLPFAOpenGLProfile, NSOpenGLProfileVersion4_1Core);
238  }
239  else
240 #endif /*MAC_OS_X_VERSION_MAX_ALLOWED*/
241  if (ctxconfig->major >= 3)
242  {
243  setAttrib(NSOpenGLPFAOpenGLProfile, NSOpenGLProfileVersion3_2Core);
244  }
245 
246  if (ctxconfig->major <= 2)
247  {
248  if (fbconfig->auxBuffers != GLFW_DONT_CARE)
249  setAttrib(NSOpenGLPFAAuxBuffers, fbconfig->auxBuffers);
250 
251  if (fbconfig->accumRedBits != GLFW_DONT_CARE &&
252  fbconfig->accumGreenBits != GLFW_DONT_CARE &&
253  fbconfig->accumBlueBits != GLFW_DONT_CARE &&
254  fbconfig->accumAlphaBits != GLFW_DONT_CARE)
255  {
256  const int accumBits = fbconfig->accumRedBits +
257  fbconfig->accumGreenBits +
258  fbconfig->accumBlueBits +
259  fbconfig->accumAlphaBits;
260 
261  setAttrib(NSOpenGLPFAAccumSize, accumBits);
262  }
263  }
264 
265  if (fbconfig->redBits != GLFW_DONT_CARE &&
266  fbconfig->greenBits != GLFW_DONT_CARE &&
267  fbconfig->blueBits != GLFW_DONT_CARE)
268  {
269  int colorBits = fbconfig->redBits +
270  fbconfig->greenBits +
271  fbconfig->blueBits;
272 
273  // macOS needs non-zero color size, so set reasonable values
274  if (colorBits == 0)
275  colorBits = 24;
276  else if (colorBits < 15)
277  colorBits = 15;
278 
279  setAttrib(NSOpenGLPFAColorSize, colorBits);
280  }
281 
282  if (fbconfig->alphaBits != GLFW_DONT_CARE)
283  setAttrib(NSOpenGLPFAAlphaSize, fbconfig->alphaBits);
284 
285  if (fbconfig->depthBits != GLFW_DONT_CARE)
286  setAttrib(NSOpenGLPFADepthSize, fbconfig->depthBits);
287 
288  if (fbconfig->stencilBits != GLFW_DONT_CARE)
289  setAttrib(NSOpenGLPFAStencilSize, fbconfig->stencilBits);
290 
291  if (fbconfig->stereo)
292  {
293 #if MAC_OS_X_VERSION_MAX_ALLOWED >= 101200
294  _glfwInputError(GLFW_FORMAT_UNAVAILABLE,
295  "NSGL: Stereo rendering is deprecated");
296  return GLFW_FALSE;
297 #else
298  addAttrib(NSOpenGLPFAStereo);
299 #endif
300  }
301 
302  if (fbconfig->doublebuffer)
303  addAttrib(NSOpenGLPFADoubleBuffer);
304 
305  if (fbconfig->samples != GLFW_DONT_CARE)
306  {
307  if (fbconfig->samples == 0)
308  {
309  setAttrib(NSOpenGLPFASampleBuffers, 0);
310  }
311  else
312  {
313  setAttrib(NSOpenGLPFASampleBuffers, 1);
314  setAttrib(NSOpenGLPFASamples, fbconfig->samples);
315  }
316  }
317 
318  // NOTE: All NSOpenGLPixelFormats on the relevant cards support sRGB
319  // framebuffer, so there's no need (and no way) to request it
320 
321  addAttrib(0);
322 
323 #undef addAttrib
324 #undef setAttrib
325 
326  window->context.nsgl.pixelFormat =
327  [[NSOpenGLPixelFormat alloc] initWithAttributes:attribs];
328  if (window->context.nsgl.pixelFormat == nil)
329  {
330  _glfwInputError(GLFW_FORMAT_UNAVAILABLE,
331  "NSGL: Failed to find a suitable pixel format");
332  return GLFW_FALSE;
333  }
334 
335  NSOpenGLContext* share = nil;
336 
337  if (ctxconfig->share)
338  share = ctxconfig->share->context.nsgl.object;
339 
340  window->context.nsgl.object =
341  [[NSOpenGLContext alloc] initWithFormat:window->context.nsgl.pixelFormat
342  shareContext:share];
343  if (window->context.nsgl.object == nil)
344  {
345  _glfwInputError(GLFW_VERSION_UNAVAILABLE,
346  "NSGL: Failed to create OpenGL context");
347  return GLFW_FALSE;
348  }
349 
350  if (fbconfig->transparent)
351  {
352  GLint opaque = 0;
353  [window->context.nsgl.object setValues:&opaque
354  forParameter:NSOpenGLContextParameterSurfaceOpacity];
355  }
356 
357  [window->ns.view setWantsBestResolutionOpenGLSurface:window->ns.retina];
358 
359  GLint interval = 0;
360  [window->context.nsgl.object setValues:&interval
361  forParameter:NSOpenGLContextParameterSwapInterval];
362 
363  [window->context.nsgl.object setView:window->ns.view];
364 
365  window->context.nsgl.swapIntervalCond = [NSCondition new];
366 
367  window->context.makeCurrent = makeContextCurrentNSGL;
368  window->context.swapBuffers = swapBuffersNSGL;
369  window->context.swapInterval = swapIntervalNSGL;
370  window->context.extensionSupported = extensionSupportedNSGL;
371  window->context.getProcAddress = getProcAddressNSGL;
372  window->context.destroy = destroyContextNSGL;
373 
374  CVDisplayLinkCreateWithActiveCGDisplays(&window->context.nsgl.displayLink);
375  CVDisplayLinkSetOutputCallback(window->context.nsgl.displayLink,
376  &displayLinkCallback,
377  window);
378  CVDisplayLinkStart(window->context.nsgl.displayLink);
379 
380  _glfwUpdateDisplayLinkDisplayNSGL(window);
381  return GLFW_TRUE;
