283 static void ImGui_ImplWGPU_SetupRenderState(ImDrawData* draw_data, WGPURenderPassEncoder ctx, FrameResources* fr)
284 {
285  // Setup orthographic projection matrix into our constant buffer
286  // Our visible imgui space lies from draw_data->DisplayPos (top left) to draw_data->DisplayPos+data_data->DisplaySize (bottom right).
287  {
288  float L = draw_data->DisplayPos.x;
289  float R = draw_data->DisplayPos.x + draw_data->DisplaySize.x;
290  float T = draw_data->DisplayPos.y;
291  float B = draw_data->DisplayPos.y + draw_data->DisplaySize.y;
292  float mvp[4][4] =
293  {
294  { 2.0f/(R-L), 0.0f, 0.0f, 0.0f },
295  { 0.0f, 2.0f/(T-B), 0.0f, 0.0f },
296  { 0.0f, 0.0f, 0.5f, 0.0f },
297  { (R+L)/(L-R), (T+B)/(B-T), 0.5f, 1.0f },
298  };
299  wgpuQueueWriteBuffer(wgpuDeviceGetDefaultQueue(g_wgpuDevice), g_resources.Uniforms, 0, mvp, sizeof(mvp));
300  }
301 
302  // Setup viewport
303  wgpuRenderPassEncoderSetViewport(ctx, 0, 0, draw_data->DisplaySize.x, draw_data->DisplaySize.y, 0, 1);
304 
305  // Bind shader and vertex buffers
306  unsigned int stride = sizeof(ImDrawVert);
307  unsigned int offset = 0;
308  wgpuRenderPassEncoderSetVertexBuffer(ctx, 0, fr->VertexBuffer, offset, fr->VertexBufferSize * stride);
309  wgpuRenderPassEncoderSetIndexBuffer(ctx, fr->IndexBuffer, sizeof(ImDrawIdx) == 2 ? WGPUIndexFormat_Uint16 : WGPUIndexFormat_Uint32, 0, fr->IndexBufferSize * sizeof(ImDrawIdx));
310  wgpuRenderPassEncoderSetPipeline(ctx, g_pipelineState);
311  wgpuRenderPassEncoderSetBindGroup(ctx, 0, g_resources.CommonBindGroup, 0, NULL);
312 
313  // Setup blend factor
314  WGPUColor blend_color = { 0.f, 0.f, 0.f, 0.f };
315  wgpuRenderPassEncoderSetBlendColor(ctx, &blend_color);
316 }

318 // Render function
319 // (this used to be set in io.RenderDrawListsFn and called by ImGui::Render(), but you can now call this directly from your main loop)
320 void ImGui_ImplWGPU_RenderDrawData(ImDrawData* draw_data, WGPURenderPassEncoder pass_encoder)
321 {
322  // Avoid rendering when minimized
323  if (draw_data->DisplaySize.x <= 0.0f || draw_data->DisplaySize.y <= 0.0f)
324  return;
325 
326  // FIXME: Assuming that this only gets called once per frame!
327  // If not, we can't just re-allocate the IB or VB, we'll have to do a proper allocator.
328  g_frameIndex = g_frameIndex + 1;
329  FrameResources* fr = &g_pFrameResources[g_frameIndex % g_numFramesInFlight];
330 
331  // Create and grow vertex/index buffers if needed
332  if (fr->VertexBuffer == NULL || fr->VertexBufferSize < draw_data->TotalVtxCount)
333  {
334  SafeRelease(fr->VertexBuffer);
335  SafeRelease(fr->VertexBufferHost);
336  fr->VertexBufferSize = draw_data->TotalVtxCount + 5000;
337 
338  WGPUBufferDescriptor vb_desc =
339  {
340  NULL,
341  "Dear ImGui Vertex buffer",
342  WGPUBufferUsage_CopyDst | WGPUBufferUsage_Vertex,
343  fr->VertexBufferSize * sizeof(ImDrawVert),
344  false
345  };
346  fr->VertexBuffer = wgpuDeviceCreateBuffer(g_wgpuDevice, &vb_desc);
347  if (!fr->VertexBuffer)
348  return;
349 
350  fr->VertexBufferHost = new ImDrawVert[fr->VertexBufferSize];
351  }
352  if (fr->IndexBuffer == NULL || fr->IndexBufferSize < draw_data->TotalIdxCount)
353  {
354  SafeRelease(fr->IndexBuffer);
355  SafeRelease(fr->IndexBufferHost);
356  fr->IndexBufferSize = draw_data->TotalIdxCount + 10000;
357 
358  WGPUBufferDescriptor ib_desc =
359  {
360  NULL,
361  "Dear ImGui Index buffer",
362  WGPUBufferUsage_CopyDst | WGPUBufferUsage_Index,
363  fr->IndexBufferSize * sizeof(ImDrawIdx),
364  false
365  };
366  fr->IndexBuffer = wgpuDeviceCreateBuffer(g_wgpuDevice, &ib_desc);
367  if (!fr->IndexBuffer)
368  return;
369 
370  fr->IndexBufferHost = new ImDrawIdx[fr->IndexBufferSize];
371  }
372 
373  // Upload vertex/index data into a single contiguous GPU buffer
374  ImDrawVert* vtx_dst = (ImDrawVert*)fr->VertexBufferHost;
375  ImDrawIdx* idx_dst = (ImDrawIdx*)fr->IndexBufferHost;
376  for (int n = 0; n < draw_data->CmdListsCount; n++)
377  {
378  const ImDrawList* cmd_list = draw_data->CmdLists[n];
379  memcpy(vtx_dst, cmd_list->VtxBuffer.Data, cmd_list->VtxBuffer.Size * sizeof(ImDrawVert));
380  memcpy(idx_dst, cmd_list->IdxBuffer.Data, cmd_list->IdxBuffer.Size * sizeof(ImDrawIdx));
381  vtx_dst += cmd_list->VtxBuffer.Size;
382  idx_dst += cmd_list->IdxBuffer.Size;
383  }
384  int64_t vb_write_size = ((char*)vtx_dst - (char*)fr->VertexBufferHost + 3) & ~3;
385  int64_t ib_write_size = ((char*)idx_dst - (char*)fr->IndexBufferHost + 3) & ~3;
386  wgpuQueueWriteBuffer(wgpuDeviceGetDefaultQueue(g_wgpuDevice), fr->VertexBuffer, 0, fr->VertexBufferHost, vb_write_size);
387  wgpuQueueWriteBuffer(wgpuDeviceGetDefaultQueue(g_wgpuDevice), fr->IndexBuffer, 0, fr->IndexBufferHost, ib_write_size);
388 
389  // Setup desired render state
390  ImGui_ImplWGPU_SetupRenderState(draw_data, pass_encoder, fr);
391 
392  // Render command lists
393  // (Because we merged all buffers into a single one, we maintain our own offset into them)
394  int global_vtx_offset = 0;
395  int global_idx_offset = 0;
396  ImVec2 clip_off = draw_data->DisplayPos;
397  for (int n = 0; n < draw_data->CmdListsCount; n++)
398  {
399  const ImDrawList* cmd_list = draw_data->CmdLists[n];
400  for (int cmd_i = 0; cmd_i < cmd_list->CmdBuffer.Size; cmd_i++)
401  {
402  const ImDrawCmd* pcmd = &cmd_list->CmdBuffer[cmd_i];
403  if (pcmd->UserCallback != NULL)
404  {
405  // User callback, registered via ImDrawList::AddCallback()
406  // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
407  if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
408  ImGui_ImplWGPU_SetupRenderState(draw_data, pass_encoder, fr);
409  else
410  pcmd->UserCallback(cmd_list, pcmd);
411  }
412  else
413  {
414  // Bind custom texture
415  auto bind_group = g_resources.ImageBindGroups.GetVoidPtr(ImHashData(&pcmd->TextureId, sizeof(ImTextureID)));
416  if (bind_group)
417  {
418  wgpuRenderPassEncoderSetBindGroup(pass_encoder, 1, (WGPUBindGroup)bind_group, 0, NULL);
419  }
420  else
421  {
422  WGPUBindGroup image_bind_group = ImGui_ImplWGPU_CreateImageBindGroup(g_resources.ImageBindGroupLayout, (WGPUTextureView)pcmd->TextureId);
423  g_resources.ImageBindGroups.SetVoidPtr(ImHashData(&pcmd->TextureId, sizeof(ImTextureID)), image_bind_group);
424  wgpuRenderPassEncoderSetBindGroup(pass_encoder, 1, image_bind_group, 0, NULL);
425  }
