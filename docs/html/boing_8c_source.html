151 /*****************************************************************************
152  * Compute a cross product (for a normal vector).
153  *
154  * c = a x b
155  *****************************************************************************/
156 void CrossProduct( vertex_t a, vertex_t b, vertex_t c, vertex_t *n )
157 {
158  GLfloat u1, u2, u3;
159  GLfloat v1, v2, v3;
160 
161  u1 = b.x - a.x;
162  u2 = b.y - a.y;
163  u3 = b.y - a.z;
164 
165  v1 = c.x - a.x;
166  v2 = c.y - a.y;
167  v3 = c.z - a.z;
168 
169  n->x = u2 * v3 - v2 * u3;
170  n->y = u3 * v1 - v3 * u1;
171  n->z = u1 * v2 - v1 * u2;
172 }

214  * reshape()
215  *****************************************************************************/
216 void reshape( GLFWwindow* window, int w, int h )
217 {
218  mat4x4 projection, view;
219 
220  glViewport( 0, 0, (GLsizei)w, (GLsizei)h );
221 
222  glMatrixMode( GL_PROJECTION );
223  mat4x4_perspective( projection,
224  2.f * (float) atan2( RADIUS, 200.f ),
225  (float)w / (float)h,
226  1.f, VIEW_SCENE_DIST );
227  glLoadMatrixf((const GLfloat*) projection);
228 
229  glMatrixMode( GL_MODELVIEW );
230  {
231  vec3 eye = { 0.f, 0.f, VIEW_SCENE_DIST };
232  vec3 center = { 0.f, 0.f, 0.f };
233  vec3 up = { 0.f, -1.f, 0.f };
234  mat4x4_look_at( view, eye, center, up );
235  }
236  glLoadMatrixf((const GLfloat*) view);
237 }

300 /*****************************************************************************
301  * Draw the Boing ball.
302  *
303  * The Boing ball is sphere in which each facet is a rectangle.
304  * Facet colors alternate between red and white.
305  * The ball is built by stacking latitudinal circles. Each circle is composed
306  * of a widely-separated set of points, so that each facet is noticeably large.
307  *****************************************************************************/
308 void DrawBoingBall( void )
309 {
310  GLfloat lon_deg; /* degree of longitude */
311  double dt_total, dt2;
312 
313  glPushMatrix();
314  glMatrixMode( GL_MODELVIEW );
315 
316  /*
317  * Another relative Z translation to separate objects.
318  */
319  glTranslatef( 0.0, 0.0, DIST_BALL );
320 
321  /* Update ball position and rotation (iterate if necessary) */
322  dt_total = dt;
323  while( dt_total > 0.0 )
324  {
325  dt2 = dt_total > MAX_DELTA_T ? MAX_DELTA_T : dt_total;
326  dt_total -= dt2;
327  BounceBall( dt2 );
328  deg_rot_y = TruncateDeg( deg_rot_y + deg_rot_y_inc*((float)dt2*ANIMATION_SPEED) );
329  }
330 
331  /* Set ball position */
332  glTranslatef( ball_x, ball_y, 0.0 );
333 
334  /*
335  * Offset the shadow.
336  */
337  if ( drawBallHow == DRAW_BALL_SHADOW )
338  {
339  glTranslatef( SHADOW_OFFSET_X,
340  SHADOW_OFFSET_Y,
341  SHADOW_OFFSET_Z );
342  }
343 
344  /*
345  * Tilt the ball.
346  */
347  glRotatef( -20.0, 0.0, 0.0, 1.0 );
348 
349  /*
350  * Continually rotate ball around Y axis.
351  */
352  glRotatef( deg_rot_y, 0.0, 1.0, 0.0 );
353 
354  /*
355  * Set OpenGL state for Boing ball.
356  */
357  glCullFace( GL_FRONT );
358  glEnable( GL_CULL_FACE );
359  glEnable( GL_NORMALIZE );
360 
361  /*
362  * Build a faceted latitude slice of the Boing ball,
363  * stepping same-sized vertical bands of the sphere.
364  */
365  for ( lon_deg = 0;
366  lon_deg < 180;
367  lon_deg += STEP_LONGITUDE )
368  {
369  /*
370  * Draw a latitude circle at this longitude.
371  */
372  DrawBoingBallBand( lon_deg,
373  lon_deg + STEP_LONGITUDE );
374  }
375 
376  glPopMatrix();
377 
378  return;
379 }

382 /*****************************************************************************
383  * Bounce the ball.
384  *****************************************************************************/
385 void BounceBall( double delta_t )
386 {
387  GLfloat sign;
388  GLfloat deg;
389 
390  if ( override_pos )
391  return;
392 
393  /* Bounce on walls */
394  if ( ball_x > (BOUNCE_WIDTH/2 + WALL_R_OFFSET ) )
395  {
396  ball_x_inc = -0.5f - 0.75f * (GLfloat)rand() / (GLfloat)RAND_MAX;
397  deg_rot_y_inc = -deg_rot_y_inc;
398  }
399  if ( ball_x < -(BOUNCE_HEIGHT/2 + WALL_L_OFFSET) )
400  {
401  ball_x_inc = 0.5f + 0.75f * (GLfloat)rand() / (GLfloat)RAND_MAX;
402  deg_rot_y_inc = -deg_rot_y_inc;
403  }
404 
405  /* Bounce on floor / roof */
406  if ( ball_y > BOUNCE_HEIGHT/2 )
407  {
408  ball_y_inc = -0.75f - 1.f * (GLfloat)rand() / (GLfloat)RAND_MAX;
409  }
410  if ( ball_y < -BOUNCE_HEIGHT/2*0.85 )
411  {
412  ball_y_inc = 0.75f + 1.f * (GLfloat)rand() / (GLfloat)RAND_MAX;
413  }
414 
415  /* Update ball position */
416  ball_x += ball_x_inc * ((float)delta_t*ANIMATION_SPEED);
417  ball_y += ball_y_inc * ((float)delta_t*ANIMATION_SPEED);
418 
419  /*
420  * Simulate the effects of gravity on Y movement.
421  */
422  if ( ball_y_inc < 0 ) sign = -1.0; else sign = 1.0;
423 
424  deg = (ball_y + BOUNCE_HEIGHT/2) * 90 / BOUNCE_HEIGHT;
425  if ( deg > 80 ) deg = 80;
426  if ( deg < 10 ) deg = 10;
427 
428  ball_y_inc = sign * 4.f * (float) sin_deg( deg );
429 }

432 /*****************************************************************************
433  * Draw a faceted latitude band of the Boing ball.
434  *
435  * Parms: long_lo, long_hi
436  * Low and high longitudes of slice, resp.
437  *****************************************************************************/
438 void DrawBoingBallBand( GLfloat long_lo,
439  GLfloat long_hi )
440 {
441  vertex_t vert_ne; /* "ne" means south-east, so on */
442  vertex_t vert_nw;
443  vertex_t vert_sw;
444  vertex_t vert_se;
445  vertex_t vert_norm;
446  GLfloat lat_deg;
447  static int colorToggle = 0;
448 
449  /*
450  * Iterate through the points of a latitude circle.
451  * A latitude circle is a 2D set of X,Z points.
452  */
453  for ( lat_deg = 0;
454  lat_deg <= (360 - STEP_LATITUDE);
455  lat_deg += STEP_LATITUDE )
456  {
457  /*
458  * Color this polygon with red or white.
459  */
460  if ( colorToggle )
461  glColor3f( 0.8f, 0.1f, 0.1f );
462  else
463  glColor3f( 0.95f, 0.95f, 0.95f );
464 #if 0
465  if ( lat_deg >= 180 )
466  if ( colorToggle )
467  glColor3f( 0.1f, 0.8f, 0.1f );
468  else
469  glColor3f( 0.5f, 0.5f, 0.95f );
470 #endif
471  colorToggle = ! colorToggle;
472 
473  /*
474  * Change color if drawing shadow.
475  */
476  if ( drawBallHow == DRAW_BALL_SHADOW )
477  glColor3f( 0.35f, 0.35f, 0.35f );
478 
479  /*
480  * Assign each Y.
481  */
482  vert_ne.y = vert_nw.y = (float) cos_deg(long_hi) * RADIUS;
483  vert_sw.y = vert_se.y = (float) cos_deg(long_lo) * RADIUS;
484 
485  /*
486  * Assign each X,Z with sin,cos values scaled by latitude radius indexed by longitude.
487  * Eg, long=0 and long=180 are at the poles, so zero scale is sin(longitude),
488  * while long=90 (sin(90)=1) is at equator.
489  */
490  vert_ne.x = (float) cos_deg( lat_deg ) * (RADIUS * (float) sin_deg( long_lo + STEP_LONGITUDE ));
491  vert_se.x = (float) cos_deg( lat_deg ) * (RADIUS * (float) sin_deg( long_lo ));
492  vert_nw.x = (float) cos_deg( lat_deg + STEP_LATITUDE ) * (RADIUS * (float) sin_deg( long_lo + STEP_LONGITUDE ));
493  vert_sw.x = (float) cos_deg( lat_deg + STEP_LATITUDE ) * (RADIUS * (float) sin_deg( long_lo ));
494 
495  vert_ne.z = (float) sin_deg( lat_deg ) * (RADIUS * (float) sin_deg( long_lo + STEP_LONGITUDE ));
496  vert_se.z = (float) sin_deg( lat_deg ) * (RADIUS * (float) sin_deg( long_lo ));
497  vert_nw.z = (float) sin_deg( lat_deg + STEP_LATITUDE ) * (RADIUS * (float) sin_deg( long_lo + STEP_LONGITUDE ));
498  vert_sw.z = (float) sin_deg( lat_deg + STEP_LATITUDE ) * (RADIUS * (float) sin_deg( long_lo ));
499 
500  /*
501  * Draw the facet.
502  */
503  glBegin( GL_POLYGON );
504 
505  CrossProduct( vert_ne, vert_nw, vert_sw, &vert_norm );
506  glNormal3f( vert_norm.x, vert_norm.y, vert_norm.z );
507 
508  glVertex3f( vert_ne.x, vert_ne.y, vert_ne.z );
509  glVertex3f( vert_nw.x, vert_nw.y, vert_nw.z );
510  glVertex3f( vert_sw.x, vert_sw.y, vert_sw.z );
511  glVertex3f( vert_se.x, vert_se.y, vert_se.z );
512 
513  glEnd();
514 
515 #if BOING_DEBUG
516  printf( "----------------------------------------------------------- \n" );
517  printf( "lat = %f long_lo = %f long_hi = %f \n", lat_deg, long_lo, long_hi );
518  printf( "vert_ne x = %.8f y = %.8f z = %.8f \n", vert_ne.x, vert_ne.y, vert_ne.z );
519  printf( "vert_nw x = %.8f y = %.8f z = %.8f \n", vert_nw.x, vert_nw.y, vert_nw.z );
520  printf( "vert_se x = %.8f y = %.8f z = %.8f \n", vert_se.x, vert_se.y, vert_se.z );
521  printf( "vert_sw x = %.8f y = %.8f z = %.8f \n", vert_sw.x, vert_sw.y, vert_sw.z );
522 #endif
523 
524  }
525 
526  /*
527  * Toggle color so that next band will opposite red/white colors than this one.
528  */
529  colorToggle = ! colorToggle;
530 
531  /*
532  * This circular band is done.
533  */
534  return;
535 }

538 /*****************************************************************************
539  * Draw the purple grid of lines, behind the Boing ball.
540  * When the Workbench is dropped to the bottom, Boing shows 12 rows.
541  *****************************************************************************/
542 void DrawGrid( void )
543 {
544  int row, col;
545  const int rowTotal = 12; /* must be divisible by 2 */
546  const int colTotal = rowTotal; /* must be same as rowTotal */
547  const GLfloat widthLine = 2.0; /* should be divisible by 2 */
548  const GLfloat sizeCell = GRID_SIZE / rowTotal;
549  const GLfloat z_offset = -40.0;
550  GLfloat xl, xr;
551  GLfloat yt, yb;
552 
553  glPushMatrix();
554  glDisable( GL_CULL_FACE );
555 
556  /*
557  * Another relative Z translation to separate objects.
558  */
559  glTranslatef( 0.0, 0.0, DIST_BALL );
560 
561  /*
562  * Draw vertical lines (as skinny 3D rectangles).
563  */
564  for ( col = 0; col <= colTotal; col++ )
565  {
566  /*
567  * Compute co-ords of line.
568  */
569  xl = -GRID_SIZE / 2 + col * sizeCell;
570  xr = xl + widthLine;
571 
572  yt = GRID_SIZE / 2;
573  yb = -GRID_SIZE / 2 - widthLine;
574 
575  glBegin( GL_POLYGON );
576 
577  glColor3f( 0.6f, 0.1f, 0.6f ); /* purple */
578 
579  glVertex3f( xr, yt, z_offset ); /* NE */
580  glVertex3f( xl, yt, z_offset ); /* NW */
581  glVertex3f( xl, yb, z_offset ); /* SW */
582  glVertex3f( xr, yb, z_offset ); /* SE */
583 
584  glEnd();
585  }
586 
587  /*
588  * Draw horizontal lines (as skinny 3D rectangles).
589  */
590  for ( row = 0; row <= rowTotal; row++ )
591  {
592  /*
593  * Compute co-ords of line.
594  */
595  yt = GRID_SIZE / 2 - row * sizeCell;
596  yb = yt - widthLine;
597 
598  xl = -GRID_SIZE / 2;
599  xr = GRID_SIZE / 2 + widthLine;
600 
601  glBegin( GL_POLYGON );
602 
603  glColor3f( 0.6f, 0.1f, 0.6f ); /* purple */
604 
605  glVertex3f( xr, yt, z_offset ); /* NE */
606  glVertex3f( xl, yt, z_offset ); /* NW */
607  glVertex3f( xl, yb, z_offset ); /* SW */
608  glVertex3f( xr, yb, z_offset ); /* SE */
609 
610  glEnd();
611  }
612 
613  glPopMatrix();
614 
615  return;
616 }

623 int main( void )
624 {
625  GLFWwindow* window;
626 
627  /* Init GLFW */
628  if( !glfwInit() )
629  exit( EXIT_FAILURE );
630 
631  window = glfwCreateWindow( 400, 400, "Boing (classic Amiga demo)", NULL, NULL );
632  if (!window)
633  {
634  glfwTerminate();
635  exit( EXIT_FAILURE );
636  }
637 
638  glfwSetWindowAspectRatio(window, 1, 1);
639 
640  glfwSetFramebufferSizeCallback(window, reshape);
641  glfwSetKeyCallback(window, key_callback);
642  glfwSetMouseButtonCallback(window, mouse_button_callback);
643  glfwSetCursorPosCallback(window, cursor_position_callback);
644 
645  glfwMakeContextCurrent(window);
646  gladLoadGL(glfwGetProcAddress);
647  glfwSwapInterval( 1 );
648 
649  glfwGetFramebufferSize(window, &width, &height);
650  reshape(window, width, height);
651 
652  glfwSetTime( 0.0 );
653 
654  init();
655 
656  /* Main loop */
657  for (;;)
658  {
659  /* Timing */
660  t = glfwGetTime();
661  dt = t - t_old;
662  t_old = t;
663 
664  /* Draw one frame */
665  display();
666 
667  /* Swap buffers */
668  glfwSwapBuffers(window);
669  glfwPollEvents();
670 
671  /* Check if we are still running */
672  if (glfwWindowShouldClose(window))
673  break;
674  }
675 
676  glfwTerminate();
677  exit( EXIT_SUCCESS );
678 }
