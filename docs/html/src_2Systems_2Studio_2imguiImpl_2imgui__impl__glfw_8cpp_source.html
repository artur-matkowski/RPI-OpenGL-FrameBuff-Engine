304 static void ImGui_ImplGlfw_UpdateMousePosAndButtons()
305 {
306  // Update buttons
307  ImGuiIO& io = ImGui::GetIO();
308  for (int i = 0; i < IM_ARRAYSIZE(io.MouseDown); i++)
309  {
310  // If a mouse press event came, always pass it as "mouse held this frame", so we don't miss click-release events that are shorter than 1 frame.
311  io.MouseDown[i] = g_MouseJustPressed[i] || glfwGetMouseButton(g_Window, i) != 0;
312  g_MouseJustPressed[i] = false;
313  }
314 
315  // Update mouse position
316  const ImVec2 mouse_pos_backup = io.MousePos;
317  io.MousePos = ImVec2(-FLT_MAX, -FLT_MAX);
318  io.MouseHoveredViewport = 0;
319  ImGuiPlatformIO& platform_io = ImGui::GetPlatformIO();
320  for (int n = 0; n < platform_io.Viewports.Size; n++)
321  {
322  ImGuiViewport* viewport = platform_io.Viewports[n];
323  GLFWwindow* window = (GLFWwindow*)viewport->PlatformHandle;
324  IM_ASSERT(window != NULL);
325 #ifdef __EMSCRIPTEN__
326  const bool focused = true;
327  IM_ASSERT(platform_io.Viewports.Size == 1);
328 #else
329  const bool focused = glfwGetWindowAttrib(window, GLFW_FOCUSED) != 0;
330 #endif
331  if (focused)
332  {
333  if (io.WantSetMousePos)
334  {
335  glfwSetCursorPos(window, (double)(mouse_pos_backup.x - viewport->Pos.x), (double)(mouse_pos_backup.y - viewport->Pos.y));
336  }
337  else
338  {
339  double mouse_x, mouse_y;
340  glfwGetCursorPos(window, &mouse_x, &mouse_y);
341  if (io.ConfigFlags & ImGuiConfigFlags_ViewportsEnable)
342  {
343  // Multi-viewport mode: mouse position in OS absolute coordinates (io.MousePos is (0,0) when the mouse is on the upper-left of the primary monitor)
344  int window_x, window_y;
345  glfwGetWindowPos(window, &window_x, &window_y);
346  io.MousePos = ImVec2((float)mouse_x + window_x, (float)mouse_y + window_y);
347  }
348  else
349  {
350  // Single viewport mode: mouse position in client window coordinates (io.MousePos is (0,0) when the mouse is on the upper-left corner of the app window)
351  io.MousePos = ImVec2((float)mouse_x, (float)mouse_y);
352  }
353  }
354  for (int i = 0; i < IM_ARRAYSIZE(io.MouseDown); i++)
355  io.MouseDown[i] |= glfwGetMouseButton(window, i) != 0;
356  }
357 
358  // (Optional) When using multiple viewports: set io.MouseHoveredViewport to the viewport the OS mouse cursor is hovering.
359  // Important: this information is not easy to provide and many high-level windowing library won't be able to provide it correctly, because
360  // - This is _ignoring_ viewports with the ImGuiViewportFlags_NoInputs flag (pass-through windows).
361  // - This is _regardless_ of whether another viewport is focused or being dragged from.
362  // If ImGuiBackendFlags_HasMouseHoveredViewport is not set by the backend, imgui will ignore this field and infer the information by relying on the
363  // rectangles and last focused time of every viewports it knows about. It will be unaware of other windows that may be sitting between or over your windows.
364  // [GLFW] FIXME: This is currently only correct on Win32. See what we do below with the WM_NCHITTEST, missing an equivalent for other systems.
365  // See https://github.com/glfw/glfw/issues/1236 if you want to help in making this a GLFW feature.
366 #if GLFW_HAS_MOUSE_PASSTHROUGH || (GLFW_HAS_WINDOW_HOVERED && defined(_WIN32))
367  const bool window_no_input = (viewport->Flags & ImGuiViewportFlags_NoInputs) != 0;
368 #if GLFW_HAS_MOUSE_PASSTHROUGH
369  glfwSetWindowAttrib(window, GLFW_MOUSE_PASSTHROUGH, window_no_input);
370 #endif
371  if (glfwGetWindowAttrib(window, GLFW_HOVERED) && !window_no_input)
372  io.MouseHoveredViewport = viewport->ID;
373 #endif
374  }
375 }

377 static void ImGui_ImplGlfw_UpdateMouseCursor()
378 {
379  ImGuiIO& io = ImGui::GetIO();
380  if ((io.ConfigFlags & ImGuiConfigFlags_NoMouseCursorChange) || glfwGetInputMode(g_Window, GLFW_CURSOR) == GLFW_CURSOR_DISABLED)
381  return;
382 
383  ImGuiMouseCursor imgui_cursor = ImGui::GetMouseCursor();
384  ImGuiPlatformIO& platform_io = ImGui::GetPlatformIO();
385  for (int n = 0; n < platform_io.Viewports.Size; n++)
386  {
387  GLFWwindow* window = (GLFWwindow*)platform_io.Viewports[n]->PlatformHandle;
388  if (imgui_cursor == ImGuiMouseCursor_None || io.MouseDrawCursor)
389  {
390  // Hide OS mouse cursor if imgui is drawing it or if it wants no cursor
391  glfwSetInputMode(window, GLFW_CURSOR, GLFW_CURSOR_HIDDEN);
392  }
393  else
394  {
395  // Show OS mouse cursor
396  // FIXME-PLATFORM: Unfocused windows seems to fail changing the mouse cursor with GLFW 3.2, but 3.3 works here.
397  glfwSetCursor(window, g_MouseCursors[imgui_cursor] ? g_MouseCursors[imgui_cursor] : g_MouseCursors[ImGuiMouseCursor_Arrow]);
398  glfwSetInputMode(window, GLFW_CURSOR, GLFW_CURSOR_NORMAL);
399  }
400  }

403 static void ImGui_ImplGlfw_UpdateGamepads()
404 {
405  ImGuiIO& io = ImGui::GetIO();
406  memset(io.NavInputs, 0, sizeof(io.NavInputs));
407  if ((io.ConfigFlags & ImGuiConfigFlags_NavEnableGamepad) == 0)
408  return;
409 
410  // Update gamepad inputs
411  #define MAP_BUTTON(NAV_NO, BUTTON_NO) { if (buttons_count > BUTTON_NO && buttons[BUTTON_NO] == GLFW_PRESS) io.NavInputs[NAV_NO] = 1.0f; }
