58 static void SetupVulkan(const char** extensions, uint32_t extensions_count)
59 {
60  VkResult err;
61 
62  // Create Vulkan Instance
63  {
64  VkInstanceCreateInfo create_info = {};
65  create_info.sType = VK_STRUCTURE_TYPE_INSTANCE_CREATE_INFO;
66  create_info.enabledExtensionCount = extensions_count;
67  create_info.ppEnabledExtensionNames = extensions;
68 #ifdef IMGUI_VULKAN_DEBUG_REPORT
69  // Enabling validation layers
70  const char* layers[] = { "VK_LAYER_KHRONOS_validation" };
71  create_info.enabledLayerCount = 1;
72  create_info.ppEnabledLayerNames = layers;
73 
74  // Enable debug report extension (we need additional storage, so we duplicate the user array to add our new extension to it)
75  const char** extensions_ext = (const char**)malloc(sizeof(const char*) * (extensions_count + 1));
76  memcpy(extensions_ext, extensions, extensions_count * sizeof(const char*));
77  extensions_ext[extensions_count] = "VK_EXT_debug_report";
78  create_info.enabledExtensionCount = extensions_count + 1;
79  create_info.ppEnabledExtensionNames = extensions_ext;
80 
81  // Create Vulkan Instance
82  err = vkCreateInstance(&create_info, g_Allocator, &g_Instance);
83  check_vk_result(err);
84  free(extensions_ext);
85 
86  // Get the function pointer (required for any extensions)
87  auto vkCreateDebugReportCallbackEXT = (PFN_vkCreateDebugReportCallbackEXT)vkGetInstanceProcAddr(g_Instance, "vkCreateDebugReportCallbackEXT");
88  IM_ASSERT(vkCreateDebugReportCallbackEXT != NULL);
89 
90  // Setup the debug report callback
91  VkDebugReportCallbackCreateInfoEXT debug_report_ci = {};
92  debug_report_ci.sType = VK_STRUCTURE_TYPE_DEBUG_REPORT_CALLBACK_CREATE_INFO_EXT;
93  debug_report_ci.flags = VK_DEBUG_REPORT_ERROR_BIT_EXT | VK_DEBUG_REPORT_WARNING_BIT_EXT | VK_DEBUG_REPORT_PERFORMANCE_WARNING_BIT_EXT;
94  debug_report_ci.pfnCallback = debug_report;
95  debug_report_ci.pUserData = NULL;
96  err = vkCreateDebugReportCallbackEXT(g_Instance, &debug_report_ci, g_Allocator, &g_DebugReport);
97  check_vk_result(err);
98 #else
99  // Create Vulkan Instance without any debug feature
100  err = vkCreateInstance(&create_info, g_Allocator, &g_Instance);
101  check_vk_result(err);
102  IM_UNUSED(g_DebugReport);
103 #endif
104  }
105 
106  // Select GPU
107  {
108  uint32_t gpu_count;
109  err = vkEnumeratePhysicalDevices(g_Instance, &gpu_count, NULL);
110  check_vk_result(err);
111  IM_ASSERT(gpu_count > 0);
112 
113  VkPhysicalDevice* gpus = (VkPhysicalDevice*)malloc(sizeof(VkPhysicalDevice) * gpu_count);
114  err = vkEnumeratePhysicalDevices(g_Instance, &gpu_count, gpus);
115  check_vk_result(err);
116 
117  // If a number >1 of GPUs got reported, you should find the best fit GPU for your purpose
118  // e.g. VK_PHYSICAL_DEVICE_TYPE_DISCRETE_GPU if available, or with the greatest memory available, etc.
119  // for sake of simplicity we'll just take the first one, assuming it has a graphics queue family.
120  g_PhysicalDevice = gpus[0];
121  free(gpus);
122  }
123 
124  // Select graphics queue family
125  {
126  uint32_t count;
127  vkGetPhysicalDeviceQueueFamilyProperties(g_PhysicalDevice, &count, NULL);
128  VkQueueFamilyProperties* queues = (VkQueueFamilyProperties*)malloc(sizeof(VkQueueFamilyProperties) * count);
129  vkGetPhysicalDeviceQueueFamilyProperties(g_PhysicalDevice, &count, queues);
130  for (uint32_t i = 0; i < count; i++)
131  if (queues[i].queueFlags & VK_QUEUE_GRAPHICS_BIT)
132  {
133  g_QueueFamily = i;
134  break;
135  }
136  free(queues);
137  IM_ASSERT(g_QueueFamily != (uint32_t)-1);
138  }
139 
140  // Create Logical Device (with 1 queue)
141  {
142  int device_extension_count = 1;
143  const char* device_extensions[] = { "VK_KHR_swapchain" };
144  const float queue_priority[] = { 1.0f };
145  VkDeviceQueueCreateInfo queue_info[1] = {};
146  queue_info[0].sType = VK_STRUCTURE_TYPE_DEVICE_QUEUE_CREATE_INFO;
147  queue_info[0].queueFamilyIndex = g_QueueFamily;
148  queue_info[0].queueCount = 1;
149  queue_info[0].pQueuePriorities = queue_priority;
150  VkDeviceCreateInfo create_info = {};
151  create_info.sType = VK_STRUCTURE_TYPE_DEVICE_CREATE_INFO;
152  create_info.queueCreateInfoCount = sizeof(queue_info) / sizeof(queue_info[0]);
153  create_info.pQueueCreateInfos = queue_info;
154  create_info.enabledExtensionCount = device_extension_count;
155  create_info.ppEnabledExtensionNames = device_extensions;
156  err = vkCreateDevice(g_PhysicalDevice, &create_info, g_Allocator, &g_Device);
157  check_vk_result(err);
158  vkGetDeviceQueue(g_Device, g_QueueFamily, 0, &g_Queue);
159  }
160 
161  // Create Descriptor Pool
162  {
163  VkDescriptorPoolSize pool_sizes[] =
164  {
165  { VK_DESCRIPTOR_TYPE_SAMPLER, 1000 },
166  { VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER, 1000 },
167  { VK_DESCRIPTOR_TYPE_SAMPLED_IMAGE, 1000 },
168  { VK_DESCRIPTOR_TYPE_STORAGE_IMAGE, 1000 },
169  { VK_DESCRIPTOR_TYPE_UNIFORM_TEXEL_BUFFER, 1000 },
170  { VK_DESCRIPTOR_TYPE_STORAGE_TEXEL_BUFFER, 1000 },
171  { VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER, 1000 },
172  { VK_DESCRIPTOR_TYPE_STORAGE_BUFFER, 1000 },
173  { VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER_DYNAMIC, 1000 },
174  { VK_DESCRIPTOR_TYPE_STORAGE_BUFFER_DYNAMIC, 1000 },
175  { VK_DESCRIPTOR_TYPE_INPUT_ATTACHMENT, 1000 }
176  };
177  VkDescriptorPoolCreateInfo pool_info = {};
178  pool_info.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_POOL_CREATE_INFO;
179  pool_info.flags = VK_DESCRIPTOR_POOL_CREATE_FREE_DESCRIPTOR_SET_BIT;
180  pool_info.maxSets = 1000 * IM_ARRAYSIZE(pool_sizes);
181  pool_info.poolSizeCount = (uint32_t)IM_ARRAYSIZE(pool_sizes);
182  pool_info.pPoolSizes = pool_sizes;
183  err = vkCreateDescriptorPool(g_Device, &pool_info, g_Allocator, &g_DescriptorPool);
184  check_vk_result(err);
185  }

241 static void FrameRender(ImGui_ImplVulkanH_Window* wd, ImDrawData* draw_data)
242 {
243  VkResult err;
244 
245  VkSemaphore image_acquired_semaphore = wd->FrameSemaphores[wd->SemaphoreIndex].ImageAcquiredSemaphore;
246  VkSemaphore render_complete_semaphore = wd->FrameSemaphores[wd->SemaphoreIndex].RenderCompleteSemaphore;
247  err = vkAcquireNextImageKHR(g_Device, wd->Swapchain, UINT64_MAX, image_acquired_semaphore, VK_NULL_HANDLE, &wd->FrameIndex);
248  if (err == VK_ERROR_OUT_OF_DATE_KHR)
249  {
250  g_SwapChainRebuild = true;
251  return;
252  }
253  check_vk_result(err);
254 
255  ImGui_ImplVulkanH_Frame* fd = &wd->Frames[wd->FrameIndex];
256  {
257  err = vkWaitForFences(g_Device, 1, &fd->Fence, VK_TRUE, UINT64_MAX); // wait indefinitely instead of periodically checking
258  check_vk_result(err);
259 
260  err = vkResetFences(g_Device, 1, &fd->Fence);
261  check_vk_result(err);
262  }
263  {
264  err = vkResetCommandPool(g_Device, fd->CommandPool, 0);
265  check_vk_result(err);
266  VkCommandBufferBeginInfo info = {};
267  info.sType = VK_STRUCTURE_TYPE_COMMAND_BUFFER_BEGIN_INFO;
268  info.flags |= VK_COMMAND_BUFFER_USAGE_ONE_TIME_SUBMIT_BIT;
269  err = vkBeginCommandBuffer(fd->CommandBuffer, &info);
270  check_vk_result(err);
271  }
272  {
273  VkRenderPassBeginInfo info = {};
274  info.sType = VK_STRUCTURE_TYPE_RENDER_PASS_BEGIN_INFO;
275  info.renderPass = wd->RenderPass;
276  info.framebuffer = fd->Framebuffer;
277  info.renderArea.extent.width = wd->Width;
278  info.renderArea.extent.height = wd->Height;
279  info.clearValueCount = 1;
280  info.pClearValues = &wd->ClearValue;
281  vkCmdBeginRenderPass(fd->CommandBuffer, &info, VK_SUBPASS_CONTENTS_INLINE);
282  }
283 
284  // Record dear imgui primitives into command buffer
285  ImGui_ImplVulkan_RenderDrawData(draw_data, fd->CommandBuffer);
286 
287  // Submit command buffer
288  vkCmdEndRenderPass(fd->CommandBuffer);
289  {
290  VkPipelineStageFlags wait_stage = VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT;
291  VkSubmitInfo info = {};
292  info.sType = VK_STRUCTURE_TYPE_SUBMIT_INFO;
293  info.waitSemaphoreCount = 1;
294  info.pWaitSemaphores = &image_acquired_semaphore;
295  info.pWaitDstStageMask = &wait_stage;
296  info.commandBufferCount = 1;
297  info.pCommandBuffers = &fd->CommandBuffer;
298  info.signalSemaphoreCount = 1;
299  info.pSignalSemaphores = &render_complete_semaphore;
300 
301  err = vkEndCommandBuffer(fd->CommandBuffer);
302  check_vk_result(err);
303  err = vkQueueSubmit(g_Queue, 1, &info, fd->Fence);
304  check_vk_result(err);
305  }
306 }

308 static void FramePresent(ImGui_ImplVulkanH_Window* wd)
309 {
310  if (g_SwapChainRebuild)
311  return;
312  VkSemaphore render_complete_semaphore = wd->FrameSemaphores[wd->SemaphoreIndex].RenderCompleteSemaphore;
313  VkPresentInfoKHR info = {};
314  info.sType = VK_STRUCTURE_TYPE_PRESENT_INFO_KHR;
315  info.waitSemaphoreCount = 1;
316  info.pWaitSemaphores = &render_complete_semaphore;
317  info.swapchainCount = 1;
318  info.pSwapchains = &wd->Swapchain;
319  info.pImageIndices = &wd->FrameIndex;
320  VkResult err = vkQueuePresentKHR(g_Queue, &info);
321  if (err == VK_ERROR_OUT_OF_DATE_KHR)
322  {
323  g_SwapChainRebuild = true;
324  return;
325  }
326  check_vk_result(err);
327  wd->SemaphoreIndex = (wd->SemaphoreIndex + 1) % wd->ImageCount; // Now we can use the next set of semaphores
328 }
