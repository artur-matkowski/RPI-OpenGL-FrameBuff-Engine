70 void init_vertices(void)
71 {
72  int x, y, p;
73 
74  // Place the vertices in a grid
75  for (y = 0; y < GRIDH; y++)
76  {
77  for (x = 0; x < GRIDW; x++)
78  {
79  p = y * GRIDW + x;
80 
81  vertex[p].x = (GLfloat) (x - GRIDW / 2) / (GLfloat) (GRIDW / 2);
82  vertex[p].y = (GLfloat) (y - GRIDH / 2) / (GLfloat) (GRIDH / 2);
83  vertex[p].z = 0;
84 
85  if ((x % 4 < 2) ^ (y % 4 < 2))
86  vertex[p].r = 0.0;
87  else
88  vertex[p].r = 1.0;
89 
90  vertex[p].g = (GLfloat) y / (GLfloat) GRIDH;
91  vertex[p].b = 1.f - ((GLfloat) x / (GLfloat) GRIDW + (GLfloat) y / (GLfloat) GRIDH) / 2.f;
92  }
93  }
94 
95  for (y = 0; y < QUADH; y++)
96  {
97  for (x = 0; x < QUADW; x++)
98  {
99  p = 4 * (y * QUADW + x);
100 
101  quad[p + 0] = y * GRIDW + x; // Some point
102  quad[p + 1] = y * GRIDW + x + 1; // Neighbor at the right side
103  quad[p + 2] = (y + 1) * GRIDW + x + 1; // Upper right neighbor
104  quad[p + 3] = (y + 1) * GRIDW + x; // Upper neighbor
105  }
106  }
107 }

118 void init_grid(void)
119 {
120  int x, y;
121  double dx, dy, d;
122 
123  for (y = 0; y < GRIDH; y++)
124  {
125  for (x = 0; x < GRIDW; x++)
126  {
127  dx = (double) (x - GRIDW / 2);
128  dy = (double) (y - GRIDH / 2);
129  d = sqrt(dx * dx + dy * dy);
130  if (d < 0.1 * (double) (GRIDW / 2))
131  {
132  d = d * 10.0;
133  p[x][y] = -cos(d * (M_PI / (double)(GRIDW * 4))) * 100.0;
134  }
135  else
136  p[x][y] = 0.0;
137 
138  vx[x][y] = 0.0;
139  vy[x][y] = 0.0;
140  }
141  }
142 }

214 //========================================================================
215 // Calculate wave propagation
216 //========================================================================
217 
218 void calc_grid(void)
219 {
220  int x, y, x2, y2;
221  double time_step = dt * ANIMATION_SPEED;
222 
223  // Compute accelerations
224  for (x = 0; x < GRIDW; x++)
225  {
226  x2 = (x + 1) % GRIDW;
227  for(y = 0; y < GRIDH; y++)
228  ax[x][y] = p[x][y] - p[x2][y];
229  }
230 
231  for (y = 0; y < GRIDH; y++)
232  {
233  y2 = (y + 1) % GRIDH;
234  for(x = 0; x < GRIDW; x++)
235  ay[x][y] = p[x][y] - p[x][y2];
236  }
237 
238  // Compute speeds
239  for (x = 0; x < GRIDW; x++)
240  {
241  for (y = 0; y < GRIDH; y++)
242  {
243  vx[x][y] = vx[x][y] + ax[x][y] * time_step;
244  vy[x][y] = vy[x][y] + ay[x][y] * time_step;
245  }
246  }
247 
248  // Compute pressure
249  for (x = 1; x < GRIDW; x++)
250  {
251  x2 = x - 1;
252  for (y = 1; y < GRIDH; y++)
253  {
254  y2 = y - 1;
255  p[x][y] = p[x][y] + (vx[x2][y] - vx[x][y] + vy[x][y2] - vy[x][y]) * time_step;
256  }
257  }
258 }
