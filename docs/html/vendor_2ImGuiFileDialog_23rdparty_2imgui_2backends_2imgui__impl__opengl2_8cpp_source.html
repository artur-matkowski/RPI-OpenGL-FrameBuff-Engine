79 static void ImGui_ImplOpenGL2_SetupRenderState(ImDrawData* draw_data, int fb_width, int fb_height)
80 {
81  // Setup render state: alpha-blending enabled, no face culling, no depth testing, scissor enabled, vertex/texcoord/color pointers, polygon fill.
82  glEnable(GL_BLEND);
83  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
84  glDisable(GL_CULL_FACE);
85  glDisable(GL_DEPTH_TEST);
86  glDisable(GL_STENCIL_TEST);
87  glDisable(GL_LIGHTING);
88  glDisable(GL_COLOR_MATERIAL);
89  glEnable(GL_SCISSOR_TEST);
90  glEnableClientState(GL_VERTEX_ARRAY);
91  glEnableClientState(GL_TEXTURE_COORD_ARRAY);
92  glEnableClientState(GL_COLOR_ARRAY);
93  glDisableClientState(GL_NORMAL_ARRAY);
94  glEnable(GL_TEXTURE_2D);
95  glPolygonMode(GL_FRONT_AND_BACK, GL_FILL);
96  glShadeModel(GL_SMOOTH);
97  glTexEnvi(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, GL_MODULATE);
98 
99  // If you are using this code with non-legacy OpenGL header/contexts (which you should not, prefer using imgui_impl_opengl3.cpp!!),
100  // you may need to backup/reset/restore other state, e.g. for current shader using the commented lines below.
101  // (DO NOT MODIFY THIS FILE! Add the code in your calling function)
102  // GLint last_program;
103  // glGetIntegerv(GL_CURRENT_PROGRAM, &last_program);
104  // glUseProgram(0);
105  // ImGui_ImplOpenGL2_RenderDrawData(...);
106  // glUseProgram(last_program)
107  // There are potentially many more states you could need to clear/setup that we can't access from default headers.
108  // e.g. glBindBuffer(GL_ARRAY_BUFFER, 0), glDisable(GL_TEXTURE_CUBE_MAP).
109 
110  // Setup viewport, orthographic projection matrix
111  // Our visible imgui space lies from draw_data->DisplayPos (top left) to draw_data->DisplayPos+data_data->DisplaySize (bottom right). DisplayPos is (0,0) for single viewport apps.
112  glViewport(0, 0, (GLsizei)fb_width, (GLsizei)fb_height);
113  glMatrixMode(GL_PROJECTION);
114  glPushMatrix();
115  glLoadIdentity();
116  glOrtho(draw_data->DisplayPos.x, draw_data->DisplayPos.x + draw_data->DisplaySize.x, draw_data->DisplayPos.y + draw_data->DisplaySize.y, draw_data->DisplayPos.y, -1.0f, +1.0f);
117  glMatrixMode(GL_MODELVIEW);
118  glPushMatrix();
119  glLoadIdentity();
120 }

122 // OpenGL2 Render function.
123 // Note that this implementation is little overcomplicated because we are saving/setting up/restoring every OpenGL state explicitly.
124 // This is in order to be able to run within an OpenGL engine that doesn't do so.
125 void ImGui_ImplOpenGL2_RenderDrawData(ImDrawData* draw_data)
126 {
127  // Avoid rendering when minimized, scale coordinates for retina displays (screen coordinates != framebuffer coordinates)
128  int fb_width = (int)(draw_data->DisplaySize.x * draw_data->FramebufferScale.x);
129  int fb_height = (int)(draw_data->DisplaySize.y * draw_data->FramebufferScale.y);
130  if (fb_width == 0 || fb_height == 0)
131  return;
132 
133  // Backup GL state
134  GLint last_texture; glGetIntegerv(GL_TEXTURE_BINDING_2D, &last_texture);
135  GLint last_polygon_mode[2]; glGetIntegerv(GL_POLYGON_MODE, last_polygon_mode);
136  GLint last_viewport[4]; glGetIntegerv(GL_VIEWPORT, last_viewport);
137  GLint last_scissor_box[4]; glGetIntegerv(GL_SCISSOR_BOX, last_scissor_box);
138  GLint last_shade_model; glGetIntegerv(GL_SHADE_MODEL, &last_shade_model);
139  GLint last_tex_env_mode; glGetTexEnviv(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, &last_tex_env_mode);
140  glPushAttrib(GL_ENABLE_BIT | GL_COLOR_BUFFER_BIT | GL_TRANSFORM_BIT);
141 
142  // Setup desired GL state
143  ImGui_ImplOpenGL2_SetupRenderState(draw_data, fb_width, fb_height);
144 
145  // Will project scissor/clipping rectangles into framebuffer space
146  ImVec2 clip_off = draw_data->DisplayPos; // (0,0) unless using multi-viewports
147  ImVec2 clip_scale = draw_data->FramebufferScale; // (1,1) unless using retina display which are often (2,2)
148 
149  // Render command lists
150  for (int n = 0; n < draw_data->CmdListsCount; n++)
151  {
152  const ImDrawList* cmd_list = draw_data->CmdLists[n];
153  const ImDrawVert* vtx_buffer = cmd_list->VtxBuffer.Data;
154  const ImDrawIdx* idx_buffer = cmd_list->IdxBuffer.Data;
155  glVertexPointer(2, GL_FLOAT, sizeof(ImDrawVert), (const GLvoid*)((const char*)vtx_buffer + IM_OFFSETOF(ImDrawVert, pos)));
156  glTexCoordPointer(2, GL_FLOAT, sizeof(ImDrawVert), (const GLvoid*)((const char*)vtx_buffer + IM_OFFSETOF(ImDrawVert, uv)));
157  glColorPointer(4, GL_UNSIGNED_BYTE, sizeof(ImDrawVert), (const GLvoid*)((const char*)vtx_buffer + IM_OFFSETOF(ImDrawVert, col)));
158 
159  for (int cmd_i = 0; cmd_i < cmd_list->CmdBuffer.Size; cmd_i++)
160  {
161  const ImDrawCmd* pcmd = &cmd_list->CmdBuffer[cmd_i];
162  if (pcmd->UserCallback)
163  {
164  // User callback, registered via ImDrawList::AddCallback()
165  // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
166  if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
167  ImGui_ImplOpenGL2_SetupRenderState(draw_data, fb_width, fb_height);
168  else
169  pcmd->UserCallback(cmd_list, pcmd);
170  }
171  else
172  {
173  // Project scissor/clipping rectangles into framebuffer space
174  ImVec4 clip_rect;
175  clip_rect.x = (pcmd->ClipRect.x - clip_off.x) * clip_scale.x;
176  clip_rect.y = (pcmd->ClipRect.y - clip_off.y) * clip_scale.y;
177  clip_rect.z = (pcmd->ClipRect.z - clip_off.x) * clip_scale.x;
178  clip_rect.w = (pcmd->ClipRect.w - clip_off.y) * clip_scale.y;
179 
180  if (clip_rect.x < fb_width && clip_rect.y < fb_height && clip_rect.z >= 0.0f && clip_rect.w >= 0.0f)
181  {
182  // Apply scissor/clipping rectangle
183  glScissor((int)clip_rect.x, (int)(fb_height - clip_rect.w), (int)(clip_rect.z - clip_rect.x), (int)(clip_rect.w - clip_rect.y));
184 
185  // Bind texture, Draw
186  glBindTexture(GL_TEXTURE_2D, (GLuint)(intptr_t)pcmd->TextureId);
187  glDrawElements(GL_TRIANGLES, (GLsizei)pcmd->ElemCount, sizeof(ImDrawIdx) == 2 ? GL_UNSIGNED_SHORT : GL_UNSIGNED_INT, idx_buffer);
188  }
189  }
190  idx_buffer += pcmd->ElemCount;
191  }
192  }
193 
194  // Restore modified GL state
195  glDisableClientState(GL_COLOR_ARRAY);
196  glDisableClientState(GL_TEXTURE_COORD_ARRAY);
197  glDisableClientState(GL_VERTEX_ARRAY);
198  glBindTexture(GL_TEXTURE_2D, (GLuint)last_texture);
199  glMatrixMode(GL_MODELVIEW);
200  glPopMatrix();
201  glMatrixMode(GL_PROJECTION);
202  glPopMatrix();
203  glPopAttrib();
204  glPolygonMode(GL_FRONT, (GLenum)last_polygon_mode[0]); glPolygonMode(GL_BACK, (GLenum)last_polygon_mode[1]);
205  glViewport(last_viewport[0], last_viewport[1], (GLsizei)last_viewport[2], (GLsizei)last_viewport[3]);
206  glScissor(last_scissor_box[0], last_scissor_box[1], (GLsizei)last_scissor_box[2], (GLsizei)last_scissor_box[3]);
207  glShadeModel(last_shade_model);
208  glTexEnvi(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, last_tex_env_mode);
209 }

211 bool ImGui_ImplOpenGL2_CreateFontsTexture()
212 {
213  // Build texture atlas
214  ImGuiIO& io = ImGui::GetIO();
215  unsigned char* pixels;
216  int width, height;
217  io.Fonts->GetTexDataAsRGBA32(&pixels, &width, &height); // Load as RGBA 32-bit (75% of the memory is wasted, but default font is so small) because it is more likely to be compatible with user's existing shaders. If your ImTextureId represent a higher-level concept than just a GL texture id, consider calling GetTexDataAsAlpha8() instead to save on GPU memory.
218 
219  // Upload texture to graphics system
220  GLint last_texture;
221  glGetIntegerv(GL_TEXTURE_BINDING_2D, &last_texture);
222  glGenTextures(1, &g_FontTexture);
223  glBindTexture(GL_TEXTURE_2D, g_FontTexture);
224  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
225  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
226  glPixelStorei(GL_UNPACK_ROW_LENGTH, 0);
227  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, width, height, 0, GL_RGBA, GL_UNSIGNED_BYTE, pixels);
228 
229  // Store our identifier
230  io.Fonts->TexID = (ImTextureID)(intptr_t)g_FontTexture;
231 
232  // Restore state
233  glBindTexture(GL_TEXTURE_2D, last_texture);
234 
235  return true;
236 }
